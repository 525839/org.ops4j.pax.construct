41  /**
42   * Support mojo inheritance by merging the local plugin metadata with metadata from dependent plugins.
43   * 
44   * Mojo inheritance is requested using a custom javadoc tag:
45   * 
46   * <code><pre>
47   *   ...
48   *   &#64;extendsPlugin archetype
49   *   &#64;goal create
50   *   ...
51   * </pre></code>
52   * 
53   * By default the current goal is taken as the goal to be extended. To extend a different goal use:
54   * 
55   * <code><pre>
56   *   ...
57   *   &#64;extendsPlugin archetype
58   *   &#64;extendsGoal create
59   *   &#64;goal create-project
60   *   ...
61   * </pre></code>
62   * 
63   * @goal inherit
64   * @phase compile
65   * @requiresDependencyResolution compile
66   */
67  public class InheritMojo extends AbstractMojo
68  {
69      /**
70       * classic mojo tag
71       */
72      private static final String GOAL = "goal";
73  
74      /**
75       * new inheritance mojo tag
76       */
77      private static final String EXTENDS_PLUGIN = "extendsPlugin";
78  
79      /**
80       * new inheritance mojo tag
81       */
82      private static final String EXTENDS_GOAL = "extendsGoal";
83  
84      /**
85       * local plugin project
86       * 
87       * @parameter expression="${project}"
88       * @readonly
89       */
90      private MavenProject m_project;
91  
92      /**
93       * output directory for the plugin project
94       * 
95       * @parameter expression="${project.build.outputDirectory}"
96       * @readonly
97       */
98      private File m_outputDirectory;
99  
100     /**
101      * support for accessing archives
102      * 
103      * @component
104      */
105     private ArchiverManager m_archiverManager;
106 
107     /**
108      * Maven plugin entry-point
109      */
110     public void execute()
111         throws MojoExecutionException
112     {
113         // pre-load available plugin metadata - local and dependencies
114         PluginXml targetPlugin = loadPluginMetadata( m_outputDirectory );
115         Map dependentPluginsByName = loadDependentPluginMetaData();
116 
117         // select maven source directories
118         JavaDocBuilder builder = new JavaDocBuilder();
119         for( Iterator i = m_project.getCompileSourceRoots().iterator(); i.hasNext(); )
120         {
121             builder.addSourceTree( new File( (String) i.next() ) );
122         }
123 
124         // scan local source for javadoc tags
125         JavaSource[] javaSources = builder.getSources();
126         for( int i = 0; i < javaSources.length; i++ )
127         {
128             JavaClass mojoClass = javaSources[i].getClasses()[0];
129 
130             // need plugin inheritance
131             DocletTag extendsTag = mojoClass.getTagByName( EXTENDS_PLUGIN );
132             if( null != extendsTag )
133             {
134                 String pluginName = extendsTag.getValue();
135                 getLog().info( "Extending " + pluginName + " plugin" );
136 
137                 // lookup using simple plugin name (ie. compiler, archetype, etc.)
138                 PluginXml superPlugin = (PluginXml) dependentPluginsByName.get( pluginName );
139                 if( null == superPlugin )
140                 {
141                     getLog().warn( pluginName + " plugin is not a dependency" );
142                 }
143                 else
144                 {
145                     mergePluginMojo( mojoClass, targetPlugin, superPlugin );
146                 }
147             }
148         }
149 
150         try
151         {
152             targetPlugin.write();
153         }
154         catch( IOException e )
155         {
156             throw new MojoExecutionException( "cannot update local plugin metadata", e );
157         }
158     }

160     /**
161      * Loads plugin metadata for the given plugin location
162      * 
163      * @param pluginDir root directory for the compiled plugin
164      * @return plugin metadata
165      * @throws MojoExecutionException
166      */
167     private PluginXml loadPluginMetadata( File pluginDir )
168         throws MojoExecutionException
169     {
170         File metadata = new File( pluginDir, "META-INF/maven/plugin.xml" );
171 
172         try
173         {
174             return new PluginXml( metadata );
175         }
176         catch( IOException e )
177         {
178             throw new MojoExecutionException( "cannot read plugin metadata " + metadata, e );
179         }
180         catch( XmlPullParserException e )
181         {
182             throw new MojoExecutionException( "cannot parse plugin metadata " + metadata, e );
183         }
184     }

186     /**
187      * Loads plugin metadata for all plugins found in this project's dependencies
188      * 
189      * @return mapping from simple plugin name to plugin metadata
190      * @throws MojoExecutionException
191      */
192     private Map loadDependentPluginMetaData()
193         throws MojoExecutionException
194     {
195         File buildArea = new File( m_project.getBuild().getDirectory(), "plugins" );
196 
197         Map pluginsByName = new HashMap();
198         for( Iterator i = m_project.getDependencyArtifacts().iterator(); i.hasNext(); )
199         {
200             Artifact artifact = (Artifact) i.next();
201             if( "maven-plugin".equals( artifact.getType() ) && artifact.getFile() != null )
202             {
203                 File unpackDir = new File( buildArea, artifact.getArtifactId() );
204                 unpackPlugin( artifact, unpackDir );
205 
206                 // extract simple plugin name by applying the standard maven naming rules in reverse
207                 String name = artifact.getArtifactId().replaceAll( "(?:maven-)?(\\w+)(?:-maven)?-plugin", "$1" );
208 
209                 pluginsByName.put( name, loadPluginMetadata( unpackDir ) );
210             }
211         }
212 
213         return pluginsByName;
214     }

216     /**
217      * Unpacks a maven plugin artifact to the given directory
218      * 
219      * @param artifact maven plugin
220      * @param unpackDir directory to unpack to
221      * @throws MojoExecutionException
222      */
223     private void unpackPlugin( Artifact artifact, File unpackDir )
224         throws MojoExecutionException
225     {
226         File pluginFile = artifact.getFile();
227         unpackDir.mkdirs();
228 
229         try
230         {
231             UnArchiver unArchiver = m_archiverManager.getUnArchiver( pluginFile );
232             unArchiver.setDestDirectory( unpackDir );
233             unArchiver.setSourceFile( pluginFile );
234             unArchiver.extract();
235         }
236         catch( NoSuchArchiverException e )
237         {
238             throw new MojoExecutionException( "cannot find unarchiver for " + pluginFile, e );
239         }
240         catch( IOException e )
241         {
242             throw new MojoExecutionException( "problem reading file " + pluginFile, e );
243         }
244         catch( ArchiverException e )
245         {
246             throw new MojoExecutionException( "problem unpacking file " + pluginFile, e );
247         }
248     }

250     /**
251      * Inherits a mojo descriptor from a dependent plugin and merge it with the local plugin metadata
252      * 
253      * @param mojoClass local mojo code requiring inheritance
254      * @param targetPlugin local plugin metadata
255      * @param superPlugin plugin metadata being extended
256      */
257     private void mergePluginMojo( JavaClass mojoClass, PluginXml targetPlugin, PluginXml superPlugin )
258     {
259         DocletTag goalTag = mojoClass.getTagByName( GOAL );
260         if( null == goalTag )
261         {
262             return;
263         }
264 
265         DocletTag superGoalTag = mojoClass.getTagByName( EXTENDS_GOAL );
266         if( null == superGoalTag )
267         {
268             superGoalTag = goalTag;
269         }
270 
271         String goal = goalTag.getValue();
272         String superGoal = superGoalTag.getValue();
273 
274         getLog().info( superGoal + " => " + goal );
275 
276         Xpp3Dom targetMojoXml = targetPlugin.findMojo( goal );
277         Xpp3Dom superMojoXml = superPlugin.findMojo( superGoal );
278 
279         PluginXml.mergeMojo( targetMojoXml, superMojoXml );
280     }
