35  /**
36   * Embed a jarfile inside a bundle project
37   * 
38   * <code><pre>
39   *   mvn pax:embed-jar [-DgroupId=...] -DartifactId=... [-Dversion=...]
40   * </pre></code>
41   * 
42   * @goal embed-jar
43   * @aggregator true
44   * 
45   * @requiresProject false
46   */
47  public class EmbedJarMojo extends AbstractMojo
48  {
49      /**
50       * Component factory for Maven artifacts
51       * 
52       * @component
53       */
54      private ArtifactFactory m_factory;
55  
56      /**
57       * Component for resolving Maven metadata
58       * 
59       * @component
60       */
61      private ArtifactMetadataSource m_source;

63      /**
64       * List of remote Maven repositories for the containing project.
65       * 
66       * @parameter expression="${project.remoteArtifactRepositories}"
67       * @required
68       * @readonly
69       */
70      private List m_remoteRepos;
71  
72      /**
73       * The local Maven repository for the containing project.
74       * 
75       * @parameter expression="${localRepository}"
76       * @required
77       * @readonly
78       */
79      private ArtifactRepository m_localRepo;
80  
81      /**
82       * The groupId of the jar to be embedded.
83       * 
84       * @parameter expression="${groupId}"
85       */
86      private String groupId;
87  
88      /**
89       * The artifactId of the jar to be embedded.
90       * 
91       * @parameter expression="${artifactId}"
92       * @required
93       */
94      private String artifactId;

97       * The version of the jar to be embedded.

101     private String version;

103     /**
104      * When true, unpack the jar inside the bundle.
105      * 
106      * @parameter expression="${unpack}"

108     private boolean unpack;

131     /**
132      * {@inheritDoc}
133      */
134     public void execute()
135         throws MojoExecutionException
136     {
137         populateMissingFields();
138 
139         updatePomDependencies();
140         updateBndInstructions();
141     }

143     /**
144      * Populate missing fields with information from the Maven repository
145      * 
146      * @throws MojoExecutionException
147      */
148     private void populateMissingFields()
149         throws MojoExecutionException
150     {
151         if( null == groupId || groupId.length() == 0 )
152         {
153             // this is a common assumption
154             groupId = artifactId;
155         }
156 
157         if( PomUtils.needReleaseVersion( version ) )
158         {
159             Artifact artifact = m_factory.createBuildArtifact( groupId, artifactId, "RELEASE", "jar" );
160             version = PomUtils.getReleaseVersion( artifact, m_source, m_remoteRepos, m_localRepo, null );
161         }
162     }
163 
164     /**
165      * Add compile-time dependency to get the jarfile, mark it optional so it's not included in transitive dependencies
166      * 
167      * @throws MojoExecutionException
168      */
169     private void updatePomDependencies()
170         throws MojoExecutionException
171     {
172         Pom pom;
173         try
174         {
175             pom = PomUtils.readPom( targetDirectory );
176         }
177         catch( IOException e )
178         {
179             throw new MojoExecutionException( "Cannot execute command."
180                 + " It requires a project with an existing pom.xml, but the build is not using one." );
181         }
182 
183         if( !pom.isBundleProject() )
184         {
185             throw new MojoExecutionException( "Cannot embed jar inside non-bundle project" );
186         }
187 
188         // new dependency to fetch the jarfile
189         Dependency dependency = new Dependency();
190         dependency.setGroupId( groupId );
191         dependency.setArtifactId( artifactId );
192         dependency.setVersion( version );
193         dependency.setScope( Artifact.SCOPE_COMPILE );
194 
195         // limit transitive nature
196         dependency.setOptional( true );
197 
198         String id = groupId + ':' + artifactId + ':' + version;
199         getLog().info( "Embedding " + id + " in " + pom );
200 
201         pom.addDependency( dependency, overwrite );
202 
203         try
204         {
205             pom.write();
206         }
207         catch( IOException e1 )
208         {
209             throw new MojoExecutionException( "Problem writing Maven POM: " + pom.getFile() );
210         }
211     }

213     /**
214      * Add Bnd instructions to embed jarfile, and update -exportcontents directive if necessary
215      * 
216      * @throws MojoExecutionException
217      */
218     private void updateBndInstructions()
219         throws MojoExecutionException
220     {
221         Bnd bnd;
222 
223         try
224         {
225             bnd = BndUtils.readBnd( targetDirectory );
226         }
227         catch( IOException e )
228         {
229             throw new MojoExecutionException( "Problem reading Bnd file: " + targetDirectory + "/osgi.bnd" );
230         }
231 
232         final String embedKey = artifactId + ";groupId=" + groupId;
233 
234         String embedDependency = bnd.getInstruction( "Embed-Dependency" );
235         embedDependency = addEmbedClause( embedKey, embedDependency );
236 
237         bnd.setInstruction( "Embed-Dependency", embedDependency, true );
238 
239         if( exportContents != null )
240         {
241             bnd.setInstruction( "-exportcontents", exportContents, overwrite );
242         }
243 
244         try
245         {
246             bnd.write();
247         }
248         catch( IOException e )
249         {
250             throw new MojoExecutionException( "Problem writing Bnd file: " + bnd.getFile() );
251         }
252     }
253 
254     /**
255      * @param embedKey partial clause identifying jarfile to embed
256      * @param embedDependency comma separated list of clauses
257      * @return updated Embed-Dependency instruction
258      */
259     private String addEmbedClause( String embedKey, String embedDependency )
260     {
261         final String embedClause = embedKey + ";inline=" + unpack;
262 
263         if( null == embedDependency )
264         {
265             return embedClause;
266         }
267 
268         StringBuffer buf = new StringBuffer();
269 
270         String[] clauses = embedDependency.split( "," );
271         for( int i = 0; i < clauses.length; i++ )
272         {
273             final String c = clauses[i].trim();
274 
275             // remove any clauses matching the one we're adding
276             if( c.length() > 0 && !c.startsWith( embedKey ) )
277             {
278                 buf.append( c );
279                 buf.append( ',' );
280             }
281         }
282 
283         // add the new clause
284         buf.append( embedClause );
285 
286         return buf.toString();
287     }
