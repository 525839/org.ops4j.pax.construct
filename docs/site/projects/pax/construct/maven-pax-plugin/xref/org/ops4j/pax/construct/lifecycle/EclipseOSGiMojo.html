104     /**
105      * {@inheritDoc}
106      */
107     public boolean setup()
108         throws MojoExecutionException
109     {
110         // we don't fork eclipse goal
111         setExecutedProject( project );
112 
113         if( null != m_provisionProject )
114         {
115             enablePDE(); // imported OSGi bundle
116         }
117         else if( PomUtils.isBundleProject( executedProject ) )
118         {
119             enablePDE(); // compiled OSGi bundle
120         }
121         else if( isProvisioningPom() )
122         {
123             try
124             {
125                 /*
126                  * unpack imported bundles and generate Eclipse files one-by-one
127                  */
128                 setupImportedBundles();
129             }
130             catch( InvalidDependencyVersionException e )
131             {
132                 getLog().warn( "Unable to generate Eclipse files for project " + executedProject.getId() );
133             }
134 
135             /*
136              * don't create Eclipse files for the provisioning POM itself!
137              */
138             return false;
139         }
140 
141         // default to normal behaviour
142         return super.setup();
143     }

145     /**
146      * Enable PDE support
147      */
148     private void enablePDE()
149     {
150         if( null == m_eclipseMojo )
151         {
152             // by default enable creation of PDE project files for OSGi
153             m_eclipseMojo = new ReflectMojo( this, EclipsePlugin.class );
154         }
155 
156         m_eclipseMojo.setField( "pde", Boolean.TRUE );
157         setWtpversion( "none" );
158     }

160     /**
161      * Does this look like a provisioning POM? ie. artifactId of 'provision', packaging type 'pom', with dependencies
162      * 
163      * @return true if this looks like a provisioning POM, otherwise false
164      */
165     private boolean isProvisioningPom()
166     {
167         // ignore POMs which don't have provision as their artifactId
168         if( !"provision".equals( executedProject.getArtifactId() ) )
169         {
170             return false;
171         }
172 
173         // ignore POMs that produce actual artifacts
174         if( !"pom".equals( executedProject.getPackaging() ) )
175         {
176             return false;
177         }
178 
179         // ignore POMs with no dependencies at all
180         List dependencies = executedProject.getDependencies();
181         if( dependencies == null || dependencies.size() == 0 )
182         {
183             return false;
184         }
185 
186         return true;
187     }

189     /**
190      * {@inheritDoc}
191      */
192     public void writeConfiguration( IdeDependency[] deps )
193         throws MojoExecutionException
194     {
195         if( !isPdeProject() )
196         {
197             // non-OSGi project
198             super.writeConfiguration( deps );
199         }
200         else
201         {
202             m_resolvedDependencies = new ArrayList();
203 
204             if( null == m_provisionProject )
205             {
206                 // compiled OSGi bundle / wrapper
207                 writeBundleConfiguration( deps );
208             }
209             else
210             {
211                 // imported (external) OSGi bundle
212                 writeImportedConfiguration();
213             }
214         }
215     }

217     /**
218      * Customize Eclipse project files for Pax-Construct generated bundles
219      * 
220      * @param deps resolved project dependencies, potentially with sources and javadocs
221      * @throws MojoExecutionException
222      */
223     private void writeBundleConfiguration( IdeDependency[] deps )
224         throws MojoExecutionException
225     {
226         for( int i = 0; i < deps.length; i++ )
227         {
228             if( deps[i].isAddedToClasspath() )
229             {
230                 /*
231                  * remove compile/runtime dependencies as we expect them to be embedded
232                  */
233                 if( !deps[i].isTestDependency() && !deps[i].isProvided() )
234                 {
235                     // cache details so we can access sources/javadocs later
236                     m_resolvedDependencies.add( deps[i] );
237                     deps[i].setAddedToClasspath( false );
238                 }
239                 else
240                 {
241                     /*
242                      * Change both test and provided dependencies to non-OSGi types otherwise they won't appear as
243                      * required libraries. We include provided dependencies here because PDE might not add them to
244                      * plugin dependencies, it all depends on the manifest which doesn't include test requirements
245                      */
246                     deps[i] = fixOSGiTestDependency( deps[i] );
247                 }
248             }
249         }
250 
251         EclipseWriterConfig config = createEclipseWriterConfig( deps );
252 
253         config.setEclipseProjectName( getEclipseProjectName( executedProject, true ) );
254 
255         new EclipseSettingsWriter().init( getLog(), config ).write();
256         new EclipseClasspathWriter().init( getLog(), config ).write();
257         new EclipseProjectWriter().init( getLog(), config ).write();
258 
259         /*
260          * copy bundle manifest to where PDE expects it, but tweak it to fix embedded paths
261          */
262         refactorForEclipse( getBundleFile( executedProject ) );
263     }

265     /**
266      * @param bundleProject bundle project
267      * @return recently built bundle
268      */
269     private File getBundleFile( MavenProject bundleProject )
270     {
271         Artifact artifact = bundleProject.getArtifact();
272         File bundleFile = artifact.getFile();
273 
274         if( null == bundleFile || !bundleFile.exists() )
275         {
276             // no file attached in this cycle, so check local build
277             String name = bundleProject.getBuild().getFinalName() + ".jar";
278             bundleFile = new File( bundleProject.getBuild().getDirectory(), name );
279         }
280 
281         if( !bundleFile.exists() )
282         {
283             // last chance: see if it is already has been installed locally
284             PomUtils.getFile( artifact, artifactResolver, localRepository );
285             bundleFile = artifact.getFile();
286         }
287 
288         return bundleFile;
289     }

291     /**
292      * The maven-eclipse-plugin has a bug where test dependencies that are also OSGi bundles are excluded from the
293      * .classpath because they are expected to be on the Bundle-ClassPath. However, this is not a valid assumption
294      * because the Maven test-cases and their dependencies do not necessarily end-up packaged inside the bundle.
295      * 
296      * @param dependency an IDE test dependency
297      * @return fixed IDE test dependency
298      */
299     private IdeDependency fixOSGiTestDependency( IdeDependency dependency )
300     {
301         // unfortunately there's no setIsOsgiBundle() method, so we have to replace the whole dependency...
302         IdeDependency testDependency = new IdeDependency( dependency.getGroupId(), dependency.getArtifactId(),
303             dependency.getVersion(), dependency.getClassifier(), dependency.isReferencedProject(), true, false, false,
304             dependency.isAddedToClasspath(), dependency.getFile(), dependency.getType(), false, null, 0 );
305 
306         testDependency.setSourceAttachment( dependency.getSourceAttachment() );
307         testDependency.setJavadocAttachment( dependency.getJavadocAttachment() );
308 
309         return testDependency;
310     }

312     /**
313      * Provide better naming for Pax-Construct generated OSGi bundles
314      * 
315      * @param project current Maven project
316      * @param addVersion when true, add the project version to the name
317      * @return an Eclipse friendly name for the bundle
318      */
319     private static String getEclipseProjectName( MavenProject project, boolean addVersion )
320     {
321         String projectName = project.getProperties().getProperty( "bundle.symbolicName" );
322         if( null == projectName )
323         {
324             // fall back to standard "groupId.artifactId" but try to eliminate duplicate segments
325             projectName = PomUtils.getCompoundId( project.getGroupId(), project.getArtifactId() );
326         }
327 
328         if( addVersion )
329         {
330             // check for wrapper version, which reflects the version of the wrapped contents
331             String projectVersion = project.getProperties().getProperty( "wrapped.version" );
332             if( null == projectVersion )
333             {
334                 projectVersion = project.getVersion();
335             }
336 
337             return projectName + " [" + projectVersion + ']';
338         }
339 
340         return projectName;
341     }

343     /**
344      * Select files for unpacking that don't exist locally under target/classes
345      */
346     private static class IncludedContentFilter
347         implements EntryFilter
348     {
349         private final File m_outputDir;
350 
351         /**
352          * @param outputDir build output directory
353          */
354         public IncludedContentFilter( File outputDir )
355         {
356             m_outputDir = outputDir;
357         }
358 
359         /**
360          * {@inheritDoc}
361          */
362         public boolean accept( String name )
363         {
364             // always select metadata folders as we may need to refactor them
365             if( name.startsWith( "META-INF" ) || name.startsWith( "OSGI-INF" ) )
366             {
367                 return true;
368             }
369 
370             // do we already have this file locally?
371             return new File( m_outputDir, name ).exists() == false;
372         }
373     }

375     /**
376      * Copy OSGi metadata to where Eclipse PDE expects it, but adjust the Bundle-ClassPath so Eclipse can find any
377      * embedded jars or directories in the unpacked bundle contents under the temporary directory
378      * 
379      * @param bundleFile the packaged bundle
380      */
381     private void refactorForEclipse( File bundleFile )
382     {
383         // temporary location in the output folder
384         String tempPath = "target/pax-eclipse";
385         boolean refactorManifest = false;
386 
387         // make relative to the provisioning POM
388         File baseDir = executedProject.getBasedir();
389         File unpackDir = new File( baseDir, tempPath );
390 
391         if( bundleFile == null || !bundleFile.exists() )
392         {
393             getLog().warn( "Bundle has not been built, reverting to basic behaviour" );
394         }
395         else
396         {
397             DirUtils.unpackBundle( bundleFile, unpackDir, new IncludedContentFilter( getBuildOutputDirectory() ) );
398 
399             moveMetadata( unpackDir, "META-INF", baseDir );
400             moveMetadata( unpackDir, "OSGI-INF", baseDir );
401 
402             // test to see if it's empty
403             unpackDir.delete();
404 
405             refactorManifest = unpackDir.exists();
406         }
407 
408         File manifestFile = new File( baseDir, "META-INF/MANIFEST.MF" );
409 
410         Manifest manifest = getBundleManifest( manifestFile );
411         Attributes mainAttributes = manifest.getMainAttributes();
412 
413         if( mainAttributes.getValue( "Bundle-SymbolicName" ) == null )
414         {
415             // Eclipse mis-behaves if the bundle has no symbolic name :(
416             String name = getEclipseProjectName( executedProject, false );
417             mainAttributes.putValue( "Bundle-SymbolicName", name.replace( '-', '_' ) );
418         }
419 
420         String bundleClassPath = mainAttributes.getValue( "Bundle-ClassPath" );
421 
422         /*
423          * refactor Bundle-ClassPath to help Eclipse find the unpacked contents
424          */
425         if( refactorManifest )
426         {
427             bundleClassPath = ".," + DirUtils.rebasePaths( bundleClassPath, tempPath, ',' );
428             mainAttributes.putValue( "Bundle-ClassPath", bundleClassPath );
429 
430             // add the embedded entries back to the Eclipse classpath
431             addEmbeddedEntriesToEclipseClassPath( tempPath, bundleClassPath );
432         }
433 
434         try
435         {
436             manifestFile.getParentFile().mkdirs();
437             FileOutputStream out = new FileOutputStream( manifestFile );
438             manifest.write( out );
439             IOUtil.close( out );
440         }
441         catch( IOException e )
442         {
443             getLog().warn( "Unable to update Eclipse manifest: " + manifestFile );
444         }
445     }

447     /**
448      * @param manifestFile path to the local bundle manifest
449      * @return the bundle manifest, or sane defaults if the manifest couldn't be opened
450      */
451     private Manifest getBundleManifest( File manifestFile )
452     {
453         Manifest manifest = new Manifest();
454 
455         try
456         {
457             FileInputStream in = new FileInputStream( manifestFile );
458             manifest.read( in );
459             IOUtil.close( in );
460         }
461         catch( IOException e )
462         {
463             Attributes mainAttributes = manifest.getMainAttributes();
464 
465             String osgiVersion = m_maven2OsgiConverter.getVersion( project.getVersion() );
466 
467             // standard OSGi entries
468             mainAttributes.putValue( "Manifest-Version", "1" );
469             mainAttributes.putValue( "Bundle-ManifestVersion", "2" );
470             mainAttributes.putValue( "Bundle-Name", project.getName() );
471             mainAttributes.putValue( "Bundle-Version", osgiVersion );
472 
473             // some basic OSGi dependencies, to help people get their code compiling...
474             mainAttributes.putValue( "Import-Package", "org.osgi.framework,org.osgi.util.tracker" );
475         }
476 
477         return manifest;
478     }

502     /**
503      * Add any embedded Bundle-ClassPath entries to the Eclipse classpath and re-attach sources/javadocs
504      * 
505      * @param bundleLocation relative path to the unpacked bundle
506      * @param bundleClassPath the refactored Bundle-ClassPath
507      */
508     private void addEmbeddedEntriesToEclipseClassPath( String bundleLocation, String bundleClassPath )
509     {
510         String[] classPath = bundleClassPath.split( "," );
511         File basedir = executedProject.getBasedir();
512 
513         try
514         {
515             File classPathFile = new File( basedir, ".classpath" );
516             Reader reader = StreamFactory.newXmlReader( classPathFile );
517             Xpp3Dom classPathXML = Xpp3DomBuilder.build( reader );
518             IOUtil.close( reader );
519 
520             for( int i = 0; i < classPath.length; i++ )
521             {
522                 String binaryPath = classPath[i].trim();
523 
524                 if( !".".equals( binaryPath ) && new File( basedir, binaryPath ).exists() )
525                 {
526                     // embedded jar/directory needs to be a 'lib' entry
527                     Xpp3Dom classPathEntry = new Xpp3Dom( "classpathentry" );
528                     classPathEntry.setAttribute( "exported", "true" );
529                     classPathEntry.setAttribute( "kind", "lib" );
530                     classPathEntry.setAttribute( "path", binaryPath );
531 
532                     // find attached sources using the previously cached IDE dependencies
533                     File sourcePath = findAttachedSource( bundleLocation, binaryPath );
534                     if( sourcePath != null )
535                     {
536                         classPathEntry.setAttribute( "sourcepath", sourcePath.getPath() );
537                     }
538 
539                     classPathXML.addChild( classPathEntry );
540                 }
541             }
542 
543             Writer writer = StreamFactory.newXmlWriter( classPathFile );
544             Xpp3DomWriter.write( new PrettyPrintXMLWriter( writer ), classPathXML );
545             IOUtil.close( writer );
546         }
547         catch( IOException e )
548         {
549             getLog().warn( "Unable to find Eclipse .classpath file" );
550         }
551         catch( XmlPullParserException e )
552         {
553             getLog().warn( "Unable to parse Eclipse .classpath file" );
554         }
555     }

557     /**
558      * Search cached IDE dependencies for an entry matching the given classpath element
559      * 
560      * @param bundleLocation relative path to the unpacked bundle
561      * @param classPathEntry classpath element
562      * @return path to the attached sources
563      */
564     private File findAttachedSource( String bundleLocation, String classPathEntry )
565     {
566         for( Iterator i = m_resolvedDependencies.iterator(); i.hasNext(); )
567         {
568             IdeDependency dependency = (IdeDependency) i.next();
569 
570             // equivalent to '.' - source is first in list
571             if( bundleLocation.equals( classPathEntry ) )
572             {
573                 return dependency.getSourceAttachment();
574             }
575             else if( Pattern.matches( "^.*[/\\\\]" + dependency.getArtifactId() + "[-.][^/\\\\]*$", classPathEntry ) )
576             {
577                 return dependency.getSourceAttachment();
578             }
579         }
580 
581         return null;
582     }

584     /**
585      * Unpack each imported bundle in turn and generate the relevant Eclipse project files
586      * 
587      * @throws InvalidDependencyVersionException
588      * @throws MojoExecutionException
589      */
590     private void setupImportedBundles()
591         throws InvalidDependencyVersionException,
592         MojoExecutionException
593     {
594         // don't process dependencies of imported bundles
595         m_provisionProject = getExecutedProject();
596         setResolveDependencies( false );
597 
598         Set artifacts = m_provisionProject.createArtifacts( artifactFactory, null, null );
599         for( Iterator i = artifacts.iterator(); i.hasNext(); )
600         {
601             Artifact artifact = (Artifact) i.next();
602 
603             // assume provided dependencies are OSGi bundles
604             if( Artifact.SCOPE_PROVIDED.equals( artifact.getScope() ) )
605             {
606                 // store project locally underneath the provisioning POM's directory
607                 File groupDir = new File( m_provisionProject.getBasedir(), "target/" + artifact.getGroupId() );
608                 File baseDir = new File( groupDir, artifact.getArtifactId() + '-' + artifact.getVersion() );
609 
610                 // download and unpack the bundle
611                 if( !PomUtils.downloadFile( artifact, artifactResolver, remoteArtifactRepositories, localRepository ) )
612                 {
613                     getLog().warn( "Skipping missing bundle " + artifact );
614                     continue;
615                 }
616 
617                 DirUtils.unpackBundle( artifact.getFile(), baseDir, null );
618 
619                 // download the bundle POM and store locally
620                 MavenProject dependencyProject = writeProjectPom( baseDir, artifact );
621                 if( null == dependencyProject )
622                 {
623                     getLog().warn( "Skipping missing bundle " + artifact );
624                     continue;
625                 }
626 
627                 setExecutedProject( dependencyProject );
628                 setProject( dependencyProject );
629 
630                 // trick Eclipse plugin to do the right thing
631                 setBuildOutputDirectory( new File( baseDir, ".ignore" ) );
632                 setEclipseProjectDir( baseDir );
633 
634                 try
635                 {
636                     // call the Eclipse plugin
637                     getLog().info( "Generating Eclipse project for bundle " + artifact );
638                     execute();
639                 }
640                 catch( MojoFailureException e )
641                 {
642                     getLog().warn( "Problem generating Eclipse files for artifact " + artifact );
643                 }
644             }
645         }
646     }

648     /**
649      * Download and save the Maven POM for the given artifact
650      * 
651      * @param baseDir base directory
652      * @param artifact Maven artifact
653      * @return the downloaded project
654      */
655     private MavenProject writeProjectPom( File baseDir, Artifact artifact )
656     {
657         MavenProject pom = null;
658 
659         String groupId = artifact.getGroupId();
660         String artifactId = artifact.getArtifactId();
661         String version = artifact.getVersion();
662 
663         Artifact pomArtifact = artifactFactory.createProjectArtifact( groupId, artifactId, version );
664 
665         try
666         {
667             pom = m_mavenProjectBuilder.buildFromRepository( pomArtifact, remoteArtifactRepositories, localRepository );
668 
669             File pomFile = new File( baseDir, "pom.xml" );
670 
671             Writer writer = StreamFactory.newXmlWriter( pomFile );
672             pom.writeModel( writer );
673             pom.setFile( pomFile );
674             IOUtil.close( writer );
675         }
676         catch( ProjectBuildingException e )
677         {
678             getLog().warn( "Unable to build POM for artifact " + pomArtifact );
679         }
680         catch( IOException e )
681         {
682             getLog().warn( "Unable to write POM for artifact " + pomArtifact );
683         }
684 
685         return pom;
686     }

688     /**
689      * Customize Eclipse project files for imported (unpacked) bundles
690      * 
691      * @throws MojoExecutionException
692      */
693     private void writeImportedConfiguration()
694         throws MojoExecutionException
695     {
696         EclipseWriterConfig config = createEclipseWriterConfig( new IdeDependency[0] );
697         config.setEclipseProjectName( getEclipseProjectName( executedProject, true ) );
698         config.setClasspathContainers( Collections.EMPTY_LIST );
699         config.setSourceDirs( new EclipseSourceDir[0] );
700 
701         // not compiling, so just need project and classpath files
702         new EclipseClasspathWriter().init( getLog(), config ).write();
703         new EclipseProjectWriter().init( getLog(), config ).write();
704 
705         Artifact sourceArtifact = artifactFactory.createArtifactWithClassifier( executedProject.getGroupId(),
706             executedProject.getArtifactId(), executedProject.getVersion(), "java-source", "sources" );
707 
708         if( downloadSources )
709         {
710             PomUtils.downloadFile( sourceArtifact, artifactResolver, remoteArtifactRepositories, localRepository );
711         }
712         else
713         {
714             // check to see if we already have the source downloaded...
715             PomUtils.getFile( sourceArtifact, artifactResolver, localRepository );
716         }
717 
718         // set PDE classpath to point to unpacked bundle
719         attachImportedContent( sourceArtifact.getFile() );
720 
721         String baseDir = executedProject.getBasedir().getPath();
722         File manifestFile = new File( baseDir, "META-INF/MANIFEST.MF" );
723 
724         Manifest manifest = getBundleManifest( manifestFile );
725         Attributes mainAttributes = manifest.getMainAttributes();
726 
727         String bundleClassPath = mainAttributes.getValue( "Bundle-ClassPath" );
728         if( null != bundleClassPath )
729         {
730             // add any embedded entries to the default Eclipse classpath
731             addEmbeddedEntriesToEclipseClassPath( baseDir, bundleClassPath );
732         }
733     }

735     /**
736      * Add a classpath entry for the unpacked imported bundle and attach it to the given source
737      * 
738      * @param sources attached bundle sources
739      */
740     private void attachImportedContent( File sources )
741     {
742         try
