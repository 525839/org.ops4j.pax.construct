35  /**
36   * Support round-trip editing of Bnd files, preserving formatting as much as possible
37   */
38  public class RoundTripBndFile
39      implements Bnd

42       * Underlying Bnd file
43       */
44      private final File m_file;

46      /**
47       * Current instructions

56      /**
57       * @param bndFile property file containing Bnd instructions
58       * @throws IOException
59       */
60      public RoundTripBndFile( File bndFile )
61          throws IOException
62      {
63          // protect against changes in working directory
64          m_file = DirUtils.resolveFile( bndFile, true );
65  
66          m_oldInstructions = new Properties();
67          m_newInstructions = new Properties();
68  
69          if( m_file.exists() )
70          {
71              FileInputStream bndStream = new FileInputStream( m_file );
72              m_oldInstructions.load( bndStream );
73              IOUtil.close( bndStream );
74          }
75  
76          m_newInstructions.putAll( m_oldInstructions );
77      }

87      /**
88       * {@inheritDoc}
89       */
90      public void setInstruction( String directive, String instruction, boolean overwrite )
91          throws ExistingInstructionException
92      {
93          if( overwrite || !m_newInstructions.containsKey( directive ) )
94          {
95              if( null == instruction )
96              {
97                  // map null instructions to the empty string
98                  m_newInstructions.setProperty( directive, "" );
99              }
100             else
101             {
102                 m_newInstructions.setProperty( directive, instruction );
103             }
104         }
105         else
106         {
107             throw new ExistingInstructionException( directive );
108         }
109     }

127     /**
128      * {@inheritDoc}
129      */
130     public void overlayInstructions( Bnd bnd )
131     {
132         for( Iterator i = bnd.getDirectives().iterator(); i.hasNext(); )
133         {
134             String directive = (String) i.next();
135             if( "Private-Package".equals( directive ) || "Export-Package".equals( directive ) )
136             {
137                 // old wrapper: maintain behaviour
138                 removeInstruction( "Embed-Dependency" );
139             }
140 
141             String instruction = bnd.getInstruction( directive );
142             m_newInstructions.setProperty( directive, instruction );
143         }
144     }

162     /**
163      * {@inheritDoc}
164      */
165     public void write()
166         throws IOException
167     {
168         if( !m_newInstructions.equals( m_oldInstructions ) || !m_file.exists() )
169         {
170             writeUpdatedInstructions();
171         }
172 
173         m_oldInstructions.clear();
174         m_oldInstructions.putAll( m_newInstructions );
175     }

177     /**
178      * Write changes to disk, preserving formatting of unaffected lines
179      * 
180      * @throws IOException
181      */
182     private void writeUpdatedInstructions()
183         throws IOException
184     {
185         List lines = readLines();
186         List block = new ArrayList();
187 
188         boolean skip = false;
189         boolean echo = true;
190 
191         Properties instructions = new Properties();
192         instructions.putAll( m_newInstructions );
193 
194         for( Iterator i = lines.iterator(); i.hasNext(); )
195         {
196             String line = (String) i.next();
197 
198             if( isWhitespaceOrComment( line ) )
199             {
200                 block.add( line );
201                 skip = false;
202             }
203             else
204             {
205                 if( !skip )
206                 {
207                     // check to see if we should update / remove / leave alone
208                     echo = checkInstructionLine( block, line, instructions );
209                 }
210 
211                 if( echo )
212                 {
213                     block.add( line );
214                 }
215 
216                 // continue skipping to end of line
217                 skip = isLineContinuation( line );
218             }
219         }
220 
221         // append any new instructions...
222         for( Enumeration e = instructions.keys(); e.hasMoreElements(); )
223         {
224             String key = (String) e.nextElement();
225             String value = instructions.getProperty( key );
226             writeInstruction( block, key, value );
227         }
228 
229         // finally write updated text back to the file
230         BufferedWriter writer = new BufferedWriter( StreamFactory.newPlatformWriter( m_file ) );
231         writeInstructionBlock( writer, block );
232         IOUtil.close( writer );
233     }

235     /**
236      * Write a block of comments and instructions to the Bnd file
237      * 
238      * @param bndWriter writer for the Bnd file
239      * @param block comment block
240      * @throws IOException
241      */
242     private static void writeInstructionBlock( BufferedWriter bndWriter, List block )
243         throws IOException
244     {
245         boolean needSpace = false;
246         for( Iterator i = block.iterator(); i.hasNext(); )
247         {
248             String line = (String) i.next();
249 
250             if( isWhitespaceOrComment( line ) )
251             {
252                 needSpace = false;
253             }
254             else if( needSpace )
255             {
256                 bndWriter.newLine();
257             }
258             else if( !isLineContinuation( line ) )
259             {
260                 needSpace = true;
261             }
262 
263             bndWriter.write( line );
264             bndWriter.newLine();
265         }
266     }

268     /**
269      * This assumes most Bnd files will be relatively small
270      * 
271      * @return list of all the lines in the Bnd file
272      * @throws IOException
273      */
274     private List readLines()
275         throws IOException
276     {
277         List lines = new ArrayList();
278 
279         if( m_file.exists() )
280         {
281             BufferedReader bndReader = new BufferedReader( StreamFactory.newPlatformReader( m_file ) );
282             while( bndReader.ready() )
283             {
284                 lines.add( bndReader.readLine() );
285             }
286             IOUtil.close( bndReader );
287         }
288 
289         return lines;
290     }

292     /**
293      * Check existing line against instructions and update if necessary
294      * 
295      * @param block comment block
296      * @param line existing line
297      * @param instructions the instructions to write to disk
298      * @return true if existing line should be echoed unchanged, otherwise false
299      */
300     private boolean checkInstructionLine( List block, String line, Properties instructions )
301     {
302         String[] keyAndValue = line.split( "[=: \t\r\n\f]", 2 );
303         String key = keyAndValue[0].trim();
304 
305         if( instructions.containsKey( key ) )
306         {
307             String newValue = (String) instructions.remove( key );
308             if( newValue.equals( m_oldInstructions.getProperty( key ) ) )
309             {
310                 // no change
311                 return true;
312             }
313 
314             // old instruction has been altered
315             writeInstruction( block, key, newValue );
316             return false;
317         }
318 
319         // remove old instruction comment
320         removeInstructionComment( block );
321         return false;
322     }

324     /**
325      * Remove the comment that's directly attached to the current instruction
326      * 
327      * @param block comment block
328      */
329     private static void removeInstructionComment( List block )
330     {
331         while( !block.isEmpty() )
332         {
333             // remove lines in reverse
334             String line = (String) block.remove( block.size() - 1 );
335 
336             // assume comment ends once we see an empty line or a non-comment
337             if( line.trim().length() == 0 || !isWhitespaceOrComment( line ) )
338             {
339                 block.add( line );
340                 return;
341             }
342         }
343     }

345     /**
346      * @param line existing line
347      * @return true if line only contains whitespace or comments
348      */
349     private static boolean isWhitespaceOrComment( String line )
350     {
351         String comment = line.trim();
352         if( comment.length() == 0 )
353         {
354             return true;
355         }
356 
357         char c = comment.charAt( 0 );
358         return '#' == c || '!' == c;
359     }

361     /**
362      * @param line existing line
363      * @return true if line ends in a continuation marker
364      */
365     private static boolean isLineContinuation( String line )
366     {
367         boolean continueLine = false;
368         for( int c = line.length() - 1; c >= 0 && '\\' == line.charAt( c ); c-- )
369         {
370             continueLine = !continueLine;
371         }
372         return continueLine;
373     }

375     /**
376      * Mark instruction clauses with line-continuation markers
377      * 
378      * @param instruction Bnd instruction
379      * @return marked instruction
380      */
381     private static String markInstructionClauses( String instruction )
382     {
383         StringBuffer buf = new StringBuffer();
384         boolean inQuotes = false;
385 
386         // add \\ markers between clauses
387         char[] text = instruction.toCharArray();
388         for( int i = 0; i < text.length; i++ )
389         {
390             char c = text[i];
391             buf.append( c );
392 
393             switch( c )
394             {
395                 case '\'':
396                 case '\"':
397                     inQuotes = !inQuotes;
398                     break;
399                 case ',':
400                     if( !inQuotes )
401                     {
402                         buf.append( '\\' );
403                     }
404                     break;
405                 default:
406                     break;
407             }
408         }
409 
410         return buf.toString();
411     }

413     /**
414      * Write instruction as a standard property, with continuation markers at every comma
415      * 
416      * @param block comment block
417      * @param key property key
418      * @param value property value
419      */
420     private static void writeInstruction( List block, String key, String value )
421     {
422         StringBuffer buf = new StringBuffer( key + ':' );
423         String instruction = markInstructionClauses( value );
424 
425         // heuristic: only wrap long instructions
426         boolean multiLine = ( instruction.length() > 80 );
427 
428         // output clauses on single or multiple lines
429         String[] clauses = instruction.split( "\\\\" );
430         for( int i = 0; i < clauses.length; i++ )
431         {
432             if( multiLine )
433             {
434                 buf.append( '\\' );
435                 block.add( buf.toString() );
436                 buf.setLength( 0 );
437                 buf.append( ' ' );
438             }
439             else if( i == 0 )
440             {
441                 buf.append( ' ' );
442             }
443 
444             buf.append( clauses[i].trim() );
445         }
446 
447         block.add( buf.toString() );
448     }
