19  import java.util.ArrayList;
20  import java.util.Arrays;
21  import java.util.HashSet;
22  import java.util.Iterator;
23  import java.util.List;
24  import java.util.Set;
25  import java.util.regex.Matcher;
26  import java.util.regex.Pattern;
27  
28  import org.codehaus.plexus.util.xml.Xpp3Dom;
29  
30  /**
31   * Provide a very simple XPATH query implementation for XML pull-parser (Xpp) documents
32   */
33  public class XppPathQuery
34  {
35      /**
36       * Node is a word
37       */
38      private static final String NODE = "\\w+";
39  
40      /**
41       * Parent is a node followed by zero or more slash-separated nodes
42       */
43      private static final String PARENT = NODE + "(?:/" + NODE + ")*";
44  
45      /**
46       * Test compares dot (ie. current node) or a node to a quoted string
47       */
48      private static final String TEST = "(.|" + NODE + ")='(.*)'";
49  
50      /**
51       * Binary operator can be and / or
52       */
53      private static final String BIN_OP = "(?:and|or)";
54  
55      /**
56       * Predicate is a test combined with zero or more tests using binary operators
57       */
58      private static final String PREDICATE = TEST + "(?:\\s+" + BIN_OP + "\\s+" + TEST + ")*";
59  
60      /**
61       * XPATH is a parent followed by a pivot node and a predicate
62       */
63      private static final String XPATH = "/?(" + PARENT + ")/(" + NODE + ")\\[\\s*(" + PREDICATE + ")\\s*\\]";
64  
65      /**
66       * Compiled XPATH expression matcher
67       */
68      private final Matcher m_xpathParser;
69  
70      /**
71       * Create a new XPATH query object from a given string
72       * 
73       * @param xpath simple XPATH query
74       * @throws IllegalArgumentException
75       */
76      public XppPathQuery( String xpath )
77          throws IllegalArgumentException
78      {
79          m_xpathParser = Pattern.compile( XPATH ).matcher( xpath );
80          if( !m_xpathParser.matches() )
81          {
82              throw new IllegalArgumentException( "Unsupported XPATH syntax: " + xpath );
83          }
84      }
85  
86      /**
87       * Find the parent node for this XPATH query
88       * 
89       * @param dom document root
90       * @return the parent node
91       */
92      public Xpp3Dom queryParent( Xpp3Dom dom )
93      {
94          String[] nodes = m_xpathParser.group( 1 ).split( "/" );
95  
96          Xpp3Dom parent = dom;
97          for( int i = 0; parent != null && i < nodes.length; i++ )
98          {
99              parent = parent.getChild( nodes[i] );
100         }
101 
102         return parent;
103     }
104 
105     /**
106      * Find all children matching the XPATH predicate
107      * 
108      * @param parent the parent node
109      * @return array of child indices
110      */
111     public int[] queryChildren( Xpp3Dom parent )
112     {
113         String pivotNode = m_xpathParser.group( 2 );
114 
115         // split into tests and binary operators
116         Pattern testPattern = Pattern.compile( TEST );
117         String[] testClauses = m_xpathParser.group( 3 ).split( "\\s+" );
118 
119         List children = new ArrayList( Arrays.asList( parent.getChildren() ) );
120 
121         Set results = new HashSet();
122         for( int i = -1; i < testClauses.length; i += 2 )
123         {
124             // parse test clause (at every even index)
125             Matcher matcher = testPattern.matcher( testClauses[i + 1] );
126             matcher.matches();
127 
128             Set selection = filter( children, pivotNode, matcher.group( 1 ), matcher.group( 2 ) );
129 
130             if( i > 0 && "and".equals( testClauses[i] ) )
131             {
132                 // and == intersect
133                 results.retainAll( selection );
134             }
135             else
136             {
137                 // or == union
138                 results.addAll( selection );
139             }
140         }
141 
142         int[] indices = new int[results.size()];
143 
144         int n = 0;
145         for( Iterator i = results.iterator(); i.hasNext(); )
146         {
147             indices[n++] = children.indexOf( i.next() );
148         }
149 
150         return indices;
151     }
152 
153     /**
154      * @param children complete list of child nodes
155      * @param pivotNode pivot node
156      * @param testNode test node
157      * @param testValue test value
158      * @return matching child nodes
159      */
160     private Set filter( List children, String pivotNode, String testNode, String testValue )
161     {
162         Set results = new HashSet();
163 
164         for( Iterator i = children.iterator(); i.hasNext(); )
165         {
166             Xpp3Dom node = (Xpp3Dom) i.next();
167             Xpp3Dom test = node;
168 
169             if( !testNode.startsWith( "." ) )
170             {
171                 test = node.getChild( testNode );
172             }
173 
174             if( pivotNode.equals( node.getName() ) && test != null && testValue.equals( test.getValue() ) )
175             {
176                 results.add( node );
177             }
178         }
179 
180         return results;
181     }
