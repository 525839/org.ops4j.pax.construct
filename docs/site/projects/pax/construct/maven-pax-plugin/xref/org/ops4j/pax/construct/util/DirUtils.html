55      /**
56       * Resolve a file to its unique canonical path - null resolves to the current directory
57       * 
58       * @param file file, may be null
59       * @param ignoreErrors ignore checked exceptions when true
60       * @return file representing the canonical path
61       */
62      public static File resolveFile( File file, boolean ignoreErrors )
63      {
64          File candidate = file;
65          if( null == file )
66          {
67              candidate = new File( "." );
68          }
69  
70          try
71          {
72              candidate = candidate.getCanonicalFile();
73          }
74          catch( IOException e )
75          {
76              if( !ignoreErrors )
77              {
78                  throw new RuntimeException( e );
79              }
80          }
81  
82          return candidate;
83      }

85      /**
86       * Search the local project tree for a Maven POM with the given id
87       * 
88       * @param baseDir directory in the project tree
89       * @param pomId either artifactId or groupId:artifactId
90       * @return a Maven POM with the given id, null if not found
91       */
92      public static Pom findPom( File baseDir, String pomId )
93      {
94          // no searching required
95          if( null == pomId || pomId.length() == 0 )
96          {
97              return null;
98          }
99  
100         // handle groupId:artifactId:other:stuff
101         String[] fields = pomId.split( ":" );
102 
103         String groupId;
104         String artifactId;
105 
106         if( fields.length > 1 )
107         {
108             groupId = fields[0];
109             artifactId = fields[1];
110         }
111         else
112         {
113             groupId = null;
114             artifactId = pomId;
115         }
116 
117         for( Iterator i = new PomIterator( baseDir ); i.hasNext(); )
118         {
119             Pom pom = (Pom) i.next();
120             if( sameProject( pom, groupId, artifactId ) )
121             {
122                 return pom;
123             }
124         }
125 
126         return null;
127     }

129     /**
130      * @param pom Maven project model
131      * @param groupId optional project group id
132      * @param artifactId project artifact id or bundle symbolic name
133      * @return true if the project has matching ids, otherwise false
134      */
135     private static boolean sameProject( Pom pom, String groupId, String artifactId )
136     {
137         if( ( artifactId.equals( pom.getArtifactId() ) || artifactId.equals( pom.getBundleSymbolicName() ) )
138             && ( null == groupId || groupId.equals( pom.getGroupId() ) ) )
139         {
140             return true;
141         }
142 
143         return false;
144     }

146     /**
147      * Verify all Maven POMs from the base directory to the target, adding missing POMs as required
148      * 
149      * @param baseDir base directory
150      * @param targetDir target directory
151      * @return the Maven project in the target directory
152      * @throws IOException
153      */
154     public static Pom createModuleTree( File baseDir, File targetDir )
155         throws IOException
156     {
157         // shortcut: target directory already has a POM
158         File pomFile = new File( targetDir, "pom.xml" );
159         if( pomFile.exists() )
160         {
161             return PomUtils.readPom( pomFile );
162         }
163 
164         String[] pivot = calculateRelativePath( baseDir, targetDir );
165         if( null == pivot )
166         {
167             // unable to find common parent directory!
168             return null;
169         }
170 
171         File commonDir = new File( pivot[1] );
172         String descentPath = pivot[2];
173 
174         Pom parentPom = null;
175         Pom childPom = null;
176 
177         int i = 0;
178         int j = -1;
179 
180         do
181         {
182             // check the next module pom...
183             String pathSoFar = descentPath.substring( 0, j + 1 );
184             pomFile = new File( commonDir, pathSoFar + "pom.xml" );
185 
186             if( pomFile.exists() )
187             {
188                 // existing pom, follow it along...
189                 childPom = PomUtils.readPom( pomFile );
190             }
191             else if( parentPom != null && "pom".equals( parentPom.getPackaging() ) )
192             {
193                 // no such pom, need to create new module pom
194                 String module = descentPath.substring( i, j );
195                 childPom = createMissingModulePom( parentPom, module, pomFile );
196             }
197             else
198             {
199                 return null; // bad project structure: cannot add interim module
200             }
201 
202             // descend to next pom
203             parentPom = childPom;
204 
205             i = j + 1;
206             j = descentPath.indexOf( '/', i );
207 
208         } while( j >= 0 );
209 
210         // final pom in target directory
211         return childPom;
212     }

214     /**
215      * Add missing Maven project POM and attach to the parent project
216      * 
217      * @param parentPom parent project
218      * @param module new project module
219      * @param pomFile new project file
220      * @return the new Maven POM
221      * @throws IOException
222      */
223     private static Pom createMissingModulePom( Pom parentPom, String module, File pomFile )
224         throws IOException
225     {
226         // link parent to new module pom
227         parentPom.addModule( module, true );
228         parentPom.write();
229 
230         String groupId = PomUtils.getCompoundId( parentPom.getGroupId(), parentPom.getArtifactId() );
231         if( groupId.equals( parentPom.getGroupId() ) )
232         {
233             groupId += '.' + module;
234         }
235 
236         // create missing module pom and link back to parent
237         Pom childPom = PomUtils.createModulePom( pomFile, groupId, module );
238         childPom.setParent( parentPom, null, true );
239         childPom.write();
240 
241         return childPom;
242     }

244     /**
245      * Calculate the relative path (and common directory) to get from a base directory to a target directory
246      * 
247      * @param baseDir base directory
248      * @param targetDir target directory
249      * @return three strings: a dotted path, absolute location of the common directory, and a descent path
250      */
251     public static String[] calculateRelativePath( File baseDir, File targetDir )
252     {
253         // need canonical form for this to work
254         File from = resolveFile( baseDir, true );
255         File to = resolveFile( targetDir, true );
256 
257         StringBuffer dottedPath = new StringBuffer();
258         StringBuffer descentPath = new StringBuffer();
259 
260         while( !from.equals( to ) )
261         {
262             // "from" is above "to", so need to descend...
263             if( from.getPath().length() < to.getPath().length() )
264             {
265                 descentPath.insert( 0, to.getName() + '/' );
266                 to = to.getParentFile();
267             }
268             // otherwise need to move up (ie "..")
269             else
270             {
271                 dottedPath.append( "../" );
272                 from = from.getParentFile();
273             }
274 
275             // reached top of file-system!
276             if( null == from || null == to )
277             {
278                 return null;
279             }
280         }
281 
282         return new String[]
283         {
284             // both "from" and "to" now hold the common directory
285             dottedPath.toString(), to.getPath(), descentPath.toString()
286         };
287     }

289     /**
290      * Set the logical parent for a given POM
291      * 
292      * @param pomFile directory or file containing the pom to update
293      * @param parentId either artifactId or groupId:artifactId
294      * @return the relative path to the parent, null if it wasn't found
295      * @throws IOException
296      */
297     public static String updateLogicalParent( File pomFile, String parentId )
298         throws IOException
299     {
300         Pom pom = PomUtils.readPom( pomFile );
301         File baseDir = pom.getBasedir();
302 
303         Pom parentPom = DirUtils.findPom( baseDir, parentId );
304         if( null == parentPom )
305         {
306             return null;
307         }
308 
309         String[] pivot = DirUtils.calculateRelativePath( baseDir, parentPom.getBasedir() );
310         if( null == pivot )
311         {
312             return null;
313         }
314 
315         String relativePath = pivot[0] + pivot[2];
316 
317         // (re-)attach project to its logical parent
318         pom.setParent( parentPom, relativePath, true );
319         pom.write();
320 
321         return relativePath;
322     }

324     /**
325      * Refactor path string, adding base directory to all entries
326      * 
327      * @param path path to refactor
328      * @param baseDir base directory to add
329      * @param pathSeparator path separator
330      * @return the refactored path
331      */
332     public static String rebasePaths( String path, String baseDir, char pathSeparator )
333     {
334         if( null == path )
335         {
336             return baseDir;
337         }
338 
339         String[] entries = path.split( Character.toString( pathSeparator ) );
340 
341         StringBuffer rebasedPath = new StringBuffer();
342         for( int i = 0; i < entries.length; i++ )
343         {
344             String pathEntry = entries[i].trim();
345 
346             if( i > 0 )
347             {
348                 rebasedPath.append( pathSeparator );
349             }
350 
351             rebasedPath.append( baseDir );
352             if( !".".equals( pathEntry ) )
353             {
354                 rebasedPath.append( '/' );
355                 rebasedPath.append( pathEntry );
356             }
357         }
358 
359         return rebasedPath.toString();
360     }

363      * Expand any bundle entries on the classpath to include embedded jars, etc.
364      * 
365      * @param outputDir current output directory
366      * @param path list of classpath elements
367      * @param tempDir temporary directory for unpacking
368      * @return expanded classpath
369      */
370     public static List expandOSGiClassPath( File outputDir, List path, File tempDir )
371     {
372         List expandedPath = new ArrayList();
373 
374         for( Iterator i = path.iterator(); i.hasNext(); )
375         {
376             File element = new File( (String) i.next() );
377             if( element.equals( outputDir ) )
378             {
379                 // don't expand the current project
380                 expandedPath.add( element.getPath() );
381             }
382             else
383             {
384                 expandedPath.addAll( expandBundleClassPath( element, tempDir ) );
385             }
386         }
387 
388         return expandedPath;
389     }

391     /**
392      * Expand compilatation classpath element to include extra entries for compiling against OSGi bundles
393      * 
394      * @param element compilatation classpath element
395      * @param tempDir temporary directory for unpacking
396      * @return expanded classpath elements
397      */
398     private static List expandBundleClassPath( File element, File tempDir )
399     {
400         File bundle = locateBundle( element );
401         if( bundle != null && bundle.isFile() )
402         {
403             String bundleClassPath = extractBundleClassPath( bundle );
404             File unpackDir = new File( tempDir, bundle.getName() );
405 
406             return unpackEmbeddedEntries( bundle, unpackDir, bundleClassPath );
407         }
408 
409         return Collections.singletonList( element.getPath() );
410     }

412     /**
413      * Locate the actual bundle for the given classpath element
414      * 
415      * @param classpathElement classpath element, may be myProject/target/classes
416      * @return the final bundle, null if it hasn't been built yet
417      */
418     private static File locateBundle( File classpathElement )
419     {
420         // assume standard output directory, ie. target/classes
421         String outputDir = "target" + File.separator + "classes";
422         String path = classpathElement.getPath();
423 
424         if( path.endsWith( outputDir ) )
425         {
426             // we need the final bundle, not the output directory, so load up the project POM
427             File projectDir = new File( path.substring( 0, path.length() - outputDir.length() ) );
428 
429             try
430             {
431                 Pom reactorPom = PomUtils.readPom( projectDir );
432                 String artifactId = reactorPom.getArtifactId();
433                 String version = reactorPom.getVersion();
434 
435                 return new File( projectDir, "target/" + artifactId + '-' + version + ".jar" );
436             }
437             catch( IOException e )
438             {
439                 return null;
440             }
441         }
442 
443         // otherwise just assume it's a bundle for now (will check for manifest later)
444         return classpathElement;
445     }

447     /**
448      * @param bundle jarfile
449      * @return Bundle-ClassPath
450      */
451     private static String extractBundleClassPath( File bundle )
452     {
453         String bundleClassPath = null;
454 
455         try
456         {
457             Manifest manifest = new JarFile( bundle ).getManifest();
458             if( null != manifest )
459             {
460                 Attributes mainAttributes = manifest.getMainAttributes();
461                 bundleClassPath = mainAttributes.getValue( "Bundle-ClassPath" );
462             }
463         }
464         catch( IOException e )
465         {
466             System.err.println( "WARNING: unable to read jarfile " + bundle );
467         }
468 
469         if( bundleClassPath != null )
470         {
471             return bundleClassPath;
472         }
473 
474         return ".";
475     }

511     /**
512      * Simple Zip unpacking code, supports selected extraction of entries
513      * 
514      * @param bundle zipfile
515      * @param here unpack directory
516      * @param filter selection filter
517      * @throws IOException
518      */
519     private static void unpack( File bundle, File here, EntryFilter filter )
520         throws IOException
521     {
522         ZipFile zipFile = new ZipFile( bundle );
523 
524         try
525         {
526             for( Enumeration e = zipFile.entries(); e.hasMoreElements(); )
527             {
528                 ZipEntry entry = (ZipEntry) e.nextElement();
529                 String name = entry.getName();
530 
531                 // don't bother with plain folders, as we always create them on-demand
532                 if( !entry.isDirectory() && ( null == filter || filter.accept( name ) ) )
533                 {
534                     // place unpacked file underneath target folder
535                     File file = FileUtils.resolveFile( here, name );
536                     file.getParentFile().mkdirs();
537 
538                     InputStream in = zipFile.getInputStream( entry );
539                     OutputStream out = new FileOutputStream( file );
540 
541                     try
542                     {
543                         // unpack contents
544                         IOUtil.copy( in, out );
545                     }
546                     finally
547                     {
548                         IOUtil.close( out );
549                         IOUtil.close( in );
550                     }
551                 }
552             }
553         }
554         finally
555         {
556             zipFile.close();
557         }
558     }

560     /**
561      * @param bundle jarfile
562      * @param here unpack directory
563      * @param bundleClassPath Bundle-ClassPath attribute
564      * @return list of paths pointing to unpacked entries
565      */
566     private static List unpackEmbeddedEntries( File bundle, File here, String bundleClassPath )
567     {
568         try
569         {
570             // improves unpacking performance
571             FileUtils.deleteDirectory( here );
572         }
573         catch( IOException e )
574         {
575             return Collections.singletonList( bundle.getPath() );
576         }
577 
578         List pathList = new ArrayList();
579         String pathPrefix = here.getPath();
580 
581         String[] entries = bundleClassPath.split( "," );
582         for( int i = 0; i < entries.length; i++ )
583         {
584             final String path = entries[i].trim();
585             if( path.length() == 0 )
586             {
587                 continue;
588             }
589             else if( ".".equals( path ) )
590             {
591                 // no need to unpack, just use jar
592                 pathList.add( bundle.getPath() );
593             }
594             else
595             {
596                 try
597                 {
598                     unpack( bundle, here, new EntryFilter()
599                     {
600                         // just unpack the embedded folder/jar
601                         public boolean accept( String entryName )
602                         {
603                             return entryName.startsWith( path );
604                         }
605                     } );
606 
607                     pathList.add( pathPrefix + '/' + path );
608                 }
609                 catch( IOException e )
610                 {
611                     continue;
612                 }
613             }
614         }
615 
616         return pathList;
617     }

619     /**
620      * Recursively delete (prune) all empty directories underneath the base directory
621      * 
622      * @param baseDir base directory
623      */
624     public static void pruneEmptyFolders( File baseDir )
625     {
626         List candidates = new ArrayList();
627         candidates.add( baseDir );
628 
629         List prunable = new ArrayList();
630         while( !candidates.isEmpty() )
631         {
632             File f = (File) candidates.remove( 0 );
633             if( f.isDirectory() )
634             {
635                 File[] files = f.listFiles();
636                 candidates.addAll( Arrays.asList( files ) );
637                 prunable.add( f );
638             }
639         }
640 
641         // delete must go in reverse
642         Collections.reverse( prunable );
643 
644         for( Iterator i = prunable.iterator(); i.hasNext(); )
645         {
646             // only deletes empty directories
647             File directory = (File) i.next();
648             directory.delete();
649         }
650     }
