198     /**
199      * {@inheritDoc}
200      */
201     public void execute()
202         throws MojoExecutionException
203     {
204         if( null == m_bundleIds )
205         {
206             m_bundleIds = new ArrayList();
207 
208             if( deployPoms != null )
209             {
210                 addAdditionalPoms();
211             }
212         }
213 
214         addBundleDependencies( m_project );
215 
216         if( m_reactorProjects.indexOf( m_project ) == m_reactorProjects.size() - 1 )
217         {
218             deployBundles();
219         }
220     }

222     /**
223      * Adds project artifact (if it's a bundle) to the deploy list as well as any provided, non-optional dependencies
224      * 
225      * @param project a Maven project
226      */
227     private void addBundleDependencies( MavenProject project )
228     {
229         if( PomUtils.isBundleProject( project ) )
230         {
231             provisionBundle( project.getArtifact() );
232         }
233 
234         try
235         {
236             Set artifacts = project.createArtifacts( m_factory, null, null );
237             for( Iterator i = artifacts.iterator(); i.hasNext(); )
238             {
239                 Artifact artifact = (Artifact) i.next();
240                 if( Artifact.SCOPE_PROVIDED.equals( artifact.getScope() ) && !artifact.isOptional() )
241                 {
242                     provisionBundle( artifact );
243                 }
244             }
245         }
246         catch( InvalidDependencyVersionException e )
247         {
248             getLog().warn( "Bad version in dependencies for " + project.getId() );
249         }
250     }

252     /**
253      * @param bundle potential bundle artifact
254      */
255     private void provisionBundle( Artifact bundle )
256     {
257         // force download here, as next check tries to avoid downloading where possible
258         if( !PomUtils.downloadFile( bundle, m_resolver, m_remoteRepos, m_localRepo ) )
259         {
260             getLog().warn( "Skipping missing artifact " + bundle );
261             return;
262         }
263 
264         if( PomUtils.isBundleArtifact( bundle, m_resolver, m_remoteRepos, m_localRepo, true ) )
265         {
266             String version = PomUtils.getMetaVersion( bundle );
267             String bundleId = bundle.getGroupId() + ':' + bundle.getArtifactId() + ':' + version;
268             if( !m_bundleIds.contains( bundleId ) )
269             {
270                 m_bundleIds.add( bundleId );
271             }
272         }
273         else
274         {
275             getLog().warn( "Skipping non-bundle artifact " + bundle );
276         }
277     }

279     /**
280      * Add user supplied POMs as if they were in the Maven reactor
281      */
282     private void addAdditionalPoms()
283     {
284         String[] pomPaths = deployPoms.split( "," );
285         for( int i = 0; i < pomPaths.length; i++ )
286         {
287             File pomFile = new File( pomPaths[i].trim() );
288             if( pomFile.exists() )
289             {
290                 try
291                 {
292                     addBundleDependencies( m_projectBuilder.build( pomFile, m_localRepo, null ) );
293                 }
294                 catch( ProjectBuildingException e )
295                 {
296                     getLog().warn( "Unable to build Maven project for " + pomFile );
297                 }
298             }
299         }
300     }

302     /**
303      * Create deployment POM and pass it onto Pax-Runner for provisioning
304      * 
305      * @throws MojoExecutionException
306      */
307     private void deployBundles()
308         throws MojoExecutionException
309     {
310         if( m_bundleIds.size() == 0 )
311         {
312             getLog().info( "~~~~~~~~~~~~~~~~~~~" );
313             getLog().info( " No bundles found! " );
314             getLog().info( "~~~~~~~~~~~~~~~~~~~" );
315         }
316 
317         List bundles = resolveProvisionedBundles();
318         MavenProject deployProject = createDeploymentProject( bundles );
319         installDeploymentPom( deployProject );
320 
321         if( !deploy )
322         {
323             getLog().info( "Deployment complete" );
324             return;
325         }
326 
327         String delim = "";
328         StringBuffer repoListBuilder = new StringBuffer();
329         for( Iterator i = m_remoteRepos.iterator(); i.hasNext(); )
330         {
331             repoListBuilder.append( delim );
332 
333             ArtifactRepository repo = (ArtifactRepository) i.next();
334             Mirror mirror = getRepositoryMirror( repo );
335             if( null == mirror )
336             {
337                 repoListBuilder.append( repo.getUrl() );
338             }
339             else
340             {
341                 repoListBuilder.append( mirror.getUrl() );
342             }
343 
344             delim = ",";
345         }
346 
347         if( PomUtils.needReleaseVersion( runner ) )
348         {
349             // find the latest release of Pax-Runner by querying the local and remote repos...
350             Artifact runnerProject = m_factory.createProjectArtifact( PAX_RUNNER_GROUP, PAX_RUNNER_ARTIFACT, runner );
351             runner = PomUtils.getReleaseVersion( runnerProject, m_source, m_remoteRepos, m_localRepo, null );
352         }
353 
354         /*
355          * Dynamically load the correct Pax-Runner code
356          */
357         if( runner.compareTo( "0.5.0" ) < 0 )
358         {
359             Class clazz = loadRunnerClass( "org.ops4j.pax", "runner", PAX_RUNNER_METHOD, false );
360             deployRunnerClassic( clazz, deployProject, repoListBuilder.toString() );
361         }
362         else
363         {
364             Class clazz = loadRunnerClass( PAX_RUNNER_GROUP, PAX_RUNNER_ARTIFACT, PAX_RUNNER_METHOD, true );
365             deployRunnerNG( clazz, deployProject, repoListBuilder.toString() );
366         }
367     }

369     /**
370      * @param repo remote Maven repository
371      * @return repository mirror, or null if it doesn't have one
372      */
373     private Mirror getRepositoryMirror( ArtifactRepository repo )
374     {
375         Mirror mirror = m_settings.getMirrorOf( repo.getId() );
376         if( null == mirror )
377         {
378             mirror = m_settings.getMirrorOf( "*" );
379         }
380         return mirror;
381     }

383     /**
384      * Attempt to resolve each provisioned bundle, and warn about any we can't find
385      * 
386      * @return list of bundles to be deployed (as Maven dependencies)
387      */
388     private List resolveProvisionedBundles()
389     {
390         List dependencies = new ArrayList();
391         for( Iterator i = m_bundleIds.iterator(); i.hasNext(); )
392         {
393             String id = (String) i.next();
394             String[] fields = id.split( ":" );
395 
396             Dependency dep = new Dependency();
397             dep.setGroupId( fields[0] );
398             dep.setArtifactId( fields[1] );
399             dep.setVersion( fields[2] );
400 
401             dependencies.add( dep );
402         }
403         return dependencies;
404     }

406     /**
407      * Create new POM (based on the root POM) which lists the deployed bundles as dependencies
408      * 
409      * @param bundles list of bundles to be deployed
410      * @return deployment project
411      * @throws MojoExecutionException
412      */
413     private MavenProject createDeploymentProject( List bundles )
414         throws MojoExecutionException
415     {
416         MavenProject rootProject = (MavenProject) m_reactorProjects.get( 0 );
417         MavenProject deployProject = new MavenProject( rootProject );
418 
419         String internalId = PomUtils.getCompoundId( rootProject.getGroupId(), rootProject.getArtifactId() );
420         deployProject.setGroupId( internalId + ".build" );
421         deployProject.setArtifactId( "deployment" );
422 
423         // remove unnecessary cruft
424         deployProject.setPackaging( "pom" );
425         deployProject.getModel().setModules( null );
426         deployProject.getModel().setDependencies( bundles );
427         deployProject.getModel().setPluginRepositories( null );
428         deployProject.getModel().setReporting( null );
429         deployProject.setBuild( null );
430 
431         File deployFile = new File( rootProject.getBasedir(), "target/deployment/pom.xml" );
432 
433         deployFile.getParentFile().mkdirs();
434         deployProject.setFile( deployFile );
435 
436         try
437         {
438             Writer writer = StreamFactory.newXmlWriter( deployFile );
439             deployProject.writeModel( writer );
440             IOUtil.close( writer );
441         }
442         catch( IOException e )
443         {
444             throw new MojoExecutionException( "Unable to write deployment POM " + deployFile );
445         }
446 
447         return deployProject;
448     }

475     /**
476      * Dynamically resolve and load the Pax-Runner class
477      * 
478      * @param groupId pax-runner group id
479      * @param artifactId pax-runner artifact id
480      * @param mainClass main pax-runner classname
481      * @param needClassifier classify pax-runner artifact according to current JVM
482      * @return main pax-runner class
483      * @throws MojoExecutionException
484      */
485     private Class loadRunnerClass( String groupId, String artifactId, String mainClass, boolean needClassifier )
486         throws MojoExecutionException
487     {
488         String jdk = null;
489         if( needClassifier && System.getProperty( "java.class.version" ).compareTo( "49.0" ) < 0 )
490         {
491             jdk = "jdk14";
492         }
493 
494         Artifact jarArtifact = m_factory.createArtifactWithClassifier( groupId, artifactId, runner, "jar", jdk );
495         if( !PomUtils.downloadFile( jarArtifact, m_resolver, m_remoteRepos, m_localRepo ) )
496         {
497             throw new MojoExecutionException( "Unable to find Pax-Runner " + jarArtifact );
498         }
499 
500         URL[] urls = new URL[1];
501         try
502         {
503             urls[0] = jarArtifact.getFile().toURI().toURL();
504         }
505         catch( MalformedURLException e )
506         {
507             throw new MojoExecutionException( "Bad Jar location " + jarArtifact.getFile() );
508         }
509 
510         try
511         {
512             ClassLoader loader = new URLClassLoader( urls );
513             Thread.currentThread().setContextClassLoader( loader );
514             return Class.forName( mainClass, true, loader );
515         }
516         catch( ClassNotFoundException e )
517         {
518             throw new MojoExecutionException( "Unable to find entry point " + mainClass + " in " + urls[0] );
519         }
520     }

522     /**
523      * Deploy bundles using the 'classic' Pax-Runner
524      * 
525      * @param mainClass main Pax-Runner class
526      * @param project deployment project
527      * @param repositories comma separated list of Maven repositories
528      * @throws MojoExecutionException
529      */
530     private void deployRunnerClassic( Class mainClass, MavenProject project, String repositories )
531         throws MojoExecutionException
532     {
533         String workDir = project.getBasedir() + "/work";
534 
535         String cachedPomName = project.getArtifactId() + '_' + project.getVersion() + ".pom";
536         File cachedPomFile = new File( workDir + "/lib/" + cachedPomName );
537 
538         // Force reload of pom
539         cachedPomFile.delete();
540 
541         String[] deployAppCmds =
542         {
543             "--dir=" + workDir, "--no-md5", "--platform=" + framework, "--profile=default",
544             "--repository=" + repositories, "--localRepository=" + m_localRepo.getBasedir(), project.getGroupId(),
545             project.getArtifactId(), project.getVersion()
546         };
547 
548         invokePaxRunner( mainClass, deployAppCmds );
549     }

551     /**
552      * Deploy bundles using the new Pax-Runner codebase
553      * 
554      * @param mainClass main Pax-Runner class
555      * @param project deployment project
556      * @param repositories comma separated list of Maven repositories
557      * @throws MojoExecutionException
558      */
559     private void deployRunnerNG( Class mainClass, MavenProject project, String repositories )
560         throws MojoExecutionException
561     {
562         String[] deployAppCmds = provision;
563 
564         String[] defaultCmds = new String[]
565         {
566             "--repositories=" + repositories, "--localRepository=" + m_localRepo.getBasedir(),
567             "--platform=" + framework, project.getFile().getAbsolutePath(), "--overwriteUserBundles"
568         };
569 
570         deployAppCmds = new String[provision.length + defaultCmds.length];
571 
572         // combine both sets of Pax-Runner settings (put defaults last)
573         System.arraycopy( provision, 0, deployAppCmds, 0, provision.length );
574         System.arraycopy( defaultCmds, 0, deployAppCmds, provision.length, defaultCmds.length );
575 
576         invokePaxRunner( mainClass, deployAppCmds );
577     }

579     /**
580      * Invoke Pax-Runner in-process
581      * 
582      * @param mainClass main Pax-Runner class
583      * @param commands array of command-line options
584      * @throws MojoExecutionException
585      */
586     private void invokePaxRunner( Class mainClass, String[] commands )
587         throws MojoExecutionException
588     {
589         Class[] paramTypes = new Class[1];
590         paramTypes[0] = String[].class;
591 
592         Object[] paramValues = new Object[1];
593         paramValues[0] = commands;
594 
595         try
596         {
597             Method entryPoint = mainClass.getMethod( "main", paramTypes );
598             entryPoint.invoke( null, paramValues );
599         }
600         catch( NoSuchMethodException e )
601         {
602             throw new MojoExecutionException( "Unable to find Pax-Runner entry point" );
603         }
604         catch( IllegalAccessException e )
605         {
606             throw new MojoExecutionException( "Unable to access Pax-Runner entry point" );
607         }
608         catch( InvocationTargetException e )
609         {
610             throw new MojoExecutionException( "Pax-Runner exception", e );
611         }
612     }
