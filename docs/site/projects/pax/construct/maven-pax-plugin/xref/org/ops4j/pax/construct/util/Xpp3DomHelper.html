17  import org.codehaus.plexus.util.xml.Xpp3Dom;

19  /**
20   * Copied from plexus-utils 1.4.2 Xpp3Dom.java (last version which appended children correctly)
21   */
22  public class Xpp3DomHelper
23  {
24      /**
25       * Merges one DOM into another, given a specific algorithm and possible override points for that algorithm.
26       * The algorithm is as follows:
27       * 
28       * 1. if the recessive DOM is null, there is nothing to do...return.
29       * 
30       * 2. Determine whether the dominant node will suppress the recessive one (flag=mergeSelf).
31       * 
32       *    A. retrieve the 'combine.self' attribute on the dominant node, and try to match against 'override'...
33       *       if it matches 'override', then set mergeSelf == false...the dominant node suppresses the recessive
34       *       one completely.
35       *       
36       *    B. otherwise, use the default value for mergeSelf, which is true...this is the same as specifying
37       *       'combine.self' == 'merge' as an attribute of the dominant root node.
38       *       
39       * 3. If mergeSelf == true
40       * 
41       *    A. if the dominant root node's value is empty, set it to the recessive root node's value
42       *    
43       *    B. For each attribute in the recessive root node which is not set in the dominant root node, set it.
44       *    
45       *    C. Determine whether children from the recessive DOM will be merged or appended to the dominant 
46       *       DOM as siblings (flag=mergeChildren).
47       *       
48       *       i.   if childMergeOverride is set (non-null), use that value (true/false)
49       *       
50       *       ii.  retrieve the 'combine.children' attribute on the dominant node, and try to match against
51       *            'append'...if it matches 'append', then set mergeChildren == false...the recessive children
52       *            will be appended as siblings of the dominant children.
53       *           
54       *       iii. otherwise, use the default value for mergeChildren, which is true...this is the same as
55       *            specifying 'combine.children' == 'merge' as an attribute on the dominant root node.
56       *    
57       *    D. Iterate through the recessive children, and:
58       *    
59       *       i.   if mergeChildren == true and there is a corresponding dominant child (matched by element name),
60       *            merge the two.
61       *            
62       *       ii.  otherwise, add the recessive child as a new child on the dominant root node.
63       */
64      private static void mergeIntoXpp3Dom( Xpp3Dom dominant, Xpp3Dom recessive, Boolean childMergeOverride )
65      {
66          // TODO: share this as some sort of assembler, implement a walk interface?
67          if ( recessive == null )
68          {
69              return;
70          }
71          
72          boolean mergeSelf = true;
73          
74          String selfMergeMode = dominant.getAttribute( Xpp3Dom.SELF_COMBINATION_MODE_ATTRIBUTE );
75          
76          if ( isNotEmpty( selfMergeMode ) && Xpp3Dom.SELF_COMBINATION_OVERRIDE.equals( selfMergeMode ) )
77          {
78              mergeSelf = false;
79          }
80          
81          if ( mergeSelf )
82          {
83              if ( isEmpty( dominant.getValue() ) )
84              {
85                  dominant.setValue( recessive.getValue() );
86              }
87              
88              String[] recessiveAttrs = recessive.getAttributeNames();
89              for ( int i = 0; i < recessiveAttrs.length; i++ )
90              {
91                  String attr = recessiveAttrs[i];
92                  
93                  if ( isEmpty( dominant.getAttribute( attr ) ) )
94                  {
95                      dominant.setAttribute( attr, recessive.getAttribute( attr ) ); 
96                  }
97              }
98              
99              boolean mergeChildren = true;
100             
101             if ( childMergeOverride != null )
102             {
103                 mergeChildren = childMergeOverride.booleanValue();
104             }
105             else
106             {
107                 String childMergeMode = dominant.getAttribute( Xpp3Dom.CHILDREN_COMBINATION_MODE_ATTRIBUTE );
108                 
109                 if ( isNotEmpty( childMergeMode ) && Xpp3Dom.CHILDREN_COMBINATION_APPEND.equals( childMergeMode ) )
110                 {
111                     mergeChildren = false;
112                 }
113             }
114             
115             Xpp3Dom[] children = recessive.getChildren();
116             for ( int i = 0; i < children.length; i++ )
117             {
118                 Xpp3Dom child = children[i];
119                 Xpp3Dom childDom = dominant.getChild( child.getName() );
120                 if ( mergeChildren && childDom != null )
121                 {
122                     mergeIntoXpp3Dom( childDom, child, childMergeOverride );
123                 }
124                 else
125                 {
126                     dominant.addChild( new Xpp3Dom( child ) );
127                 }
128             }
129         }
130     }

132     /**
133      * Merge two DOMs, with one having dominance in the case of collision.
134      * 
135      * @see #CHILDREN_COMBINATION_MODE_ATTRIBUTE
136      * @see #SELF_COMBINATION_MODE_ATTRIBUTE
137      * 
138      * @param dominant The dominant DOM into which the recessive value/attributes/children will be merged
139      * @param recessive The recessive DOM, which will be merged into the dominant DOM
140      * @param childMergeOverride Overrides attribute flags to force merging or appending of child elements
141      *        into the dominant DOM
142      */
143     public static Xpp3Dom mergeXpp3Dom( Xpp3Dom dominant, Xpp3Dom recessive, Boolean childMergeOverride )
144     {
145         if ( dominant != null )
146         {
147             mergeIntoXpp3Dom( dominant, recessive, childMergeOverride );
148             return dominant;
149         }
150         return recessive;
151     }

153     /**
154      * Merge two DOMs, with one having dominance in the case of collision.
155      * Merge mechanisms (vs. override for nodes, or vs. append for children) is determined by
156      * attributes of the dominant root node.
157      * 
158      * @see #CHILDREN_COMBINATION_MODE_ATTRIBUTE
159      * @see #SELF_COMBINATION_MODE_ATTRIBUTE
160      * 
161      * @param dominant The dominant DOM into which the recessive value/attributes/children will be merged
162      * @param recessive The recessive DOM, which will be merged into the dominant DOM
163      */
164     public static Xpp3Dom mergeXpp3Dom( Xpp3Dom dominant, Xpp3Dom recessive )
165     {
166         if ( dominant != null )
167         {
168             mergeIntoXpp3Dom( dominant, recessive, null );
169             return dominant;
170         }
171         return recessive;
172     }
