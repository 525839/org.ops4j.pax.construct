71      /**
72       * Create a new archetype fragment
73       * 
74       * @param tempDir some temporary directory
75       * @param namespace primary namespace, may be null
76       */
77      public ArchetypeFragment( File tempDir, String namespace )
78      {
79          // always allow partial use
80          m_model = new ArchetypeModel();
81          m_model.setAllowPartial( true );
82  
83          // primary Java package
84          m_namespace = namespace;
85  
86          // unique scratch directory for the fragment assembly
87          m_tempDir = new File( tempDir, "fragment" + ( m_fragmentCount++ ) );
88      }

98          File to = new File( m_tempDir, "archetype-resources" );
99  
100         File baseDir;
101         if( null == pom )
102         {

104         }
105         else

114     /**
115      * Add collection of source files
116      * 
117      * @param projectDir project directory
118      * @param path source location
119      * @param isTest true if these are test sources, otherwise false
120      */
121     public void addSources( File projectDir, String path, boolean isTest )
122     {
123         String[] pivot = DirUtils.calculateRelativePath( projectDir, new File( path ) );
124         if( null == pivot || pivot[2].length() == 0 )
125         {
126             return;
127         }
128 
129         // use relative path in search
130         String sourcePath = pivot[2];
131 
132         // primary source location
133         String packagePath = sourcePath + m_namespace.replace( '.', '/' ) + '/';
134 
135         File to = new File( m_tempDir, "archetype-resources" );
136         for( Iterator i = getFilenames( projectDir, sourcePath, null, null ).iterator(); i.hasNext(); )
137         {
138             String filename = (String) i.next();
139 
140             // relocate to 'classic' archetype location (primary package gets trimmed)
141             String target = StringUtils.replace( filename, packagePath, sourcePath );
142             int status = translateFile( projectDir, filename, to, target );
143             if( NO_SUCH_FILE == status )
144             {
145                 continue;
146             }
147             else if( target.equals( filename ) || BINARY_FILE == status )
148             {
149                 // either no relocation (or no filtering) is required
150                 addResourceEntry( filename, isTest, TEXT_FILE == status );
151             }
152             else
153             {
154                 addSourceEntry( target, isTest );
155             }
156         }
157     }

171     /**
172      * Add collection of resource files
173      * 
174      * @param projectDir project directory
175      * @param path resource location
176      * @param includes list of included names
177      * @param excludes list of excluded names
178      * @param isTest true if these are test resources, otherwise false
179      */
180     public void addResources( File projectDir, String path, List includes, List excludes, boolean isTest )
181     {
182         String[] pivot = DirUtils.calculateRelativePath( projectDir, new File( path ) );
183         if( null == pivot )
184         {
185             return;
186         }
187 
188         // use relative path in search
189         String resourcePath = pivot[2];
190 
191         File to = new File( m_tempDir, "archetype-resources" );
192         for( Iterator i = getFilenames( projectDir, resourcePath, includes, excludes ).iterator(); i.hasNext(); )
193         {
194             String filename = (String) i.next();
195 
196             // special case for Bnd instructions
197             String target = filename;
198             if( target.endsWith( ".bnd" ) || target.endsWith( "osgi.bundle" ) )
199             {
200                 target = "osgi.bnd";
201             }
202 
203             // relocate to 'classic' archetype location
204             int status = translateFile( projectDir, filename, to, target );
205             if( NO_SUCH_FILE != status )
206             {
207                 addResourceEntry( target, isTest, TEXT_FILE == status );
208             }
209         }
210     }

212     /**
213      * Get all filenames for a given location, ignoring Maven build output directories
214      * 
215      * @param dir project directory
216      * @param path location in project
217      * @param includes list of included names
218      * @param excludes list of excluded names
219      * @return list of filenames
220      */
221     private static List getFilenames( File dir, String path, List includes, List excludes )
222     {
223         DirectoryScanner scanner = new DirectoryScanner();
224 
225         String[] pathExclude = parseFilter( path, excludes, "**/target/" );
226         String[] pathInclude = parseFilter( path, includes, "**" );
227 
228         scanner.setExcludes( pathExclude );
229         scanner.setIncludes( pathInclude );
230         scanner.setFollowSymlinks( false );
231         scanner.addDefaultExcludes();
232         scanner.setBasedir( dir );
233 
234         scanner.scan();
235 
236         List filenames = new ArrayList();
237 
238         String[] includedFiles = scanner.getIncludedFiles();
239         for( int i = 0; i < includedFiles.length; i++ )
240         {
241             // normalize to use the standard Maven file separator (same as UNIX)
242             filenames.add( includedFiles[i].replace( File.separatorChar, '/' ) );
243         }
244 
245         return filenames;
246     }

248     /**
249      * @param path directory path
250      * @param filters list of ant-style filters
251      * @param defaultFilter default filter
252      * @return array of filters for the directory path
253      */
254     private static String[] parseFilter( String path, List filters, String defaultFilter )
255     {
256         String[] filterArray;
257 
258         if( null == filters || filters.size() == 0 )
259         {
260             filterArray = new String[1];
261             filterArray[0] = path + defaultFilter;
262         }
263         else
264         {
265             filterArray = (String[]) filters.toArray( new String[filters.size()] );
266             for( int i = 0; i < filterArray.length; i++ )
267             {
268                 filterArray[i] = path + filterArray[i];
269             }
270         }
271 
272         return filterArray;
273     }

312     /**
313      * Install archetype fragment to the local repository under the given build artifact
314      * 
315      * @param artifact build artifact
316      * @param archiver Jar archiver
317      * @param installer Maven artifact installer
318      * @param repo local Maven repository
319      * @throws MojoExecutionException
320      */
321     public void install( Artifact artifact, Archiver archiver, ArtifactInstaller installer, ArtifactRepository repo )
322         throws MojoExecutionException
323     {
324         m_model.setId( artifact.getGroupId() + '.' + artifact.getArtifactId() );
325         File modelFile = new File( m_tempDir, "META-INF/archetype.xml" );
326         modelFile.getParentFile().mkdirs();
327 
328         try
329         {
330             m_model.write( modelFile );
331         }
332         catch( IOException e )
333         {
334             throw new MojoExecutionException( "I/O error saving archetype model", e );
335         }
336 
337         // mimic directory structure in local repository
338         File jarFile = new File( m_tempDir.getParentFile(), repo.pathOf( artifact ) );
339         jarFile.getParentFile().mkdirs();
340 
341         try
342         {
343             // archive fragment files
344             archiver.setDestFile( jarFile );
345             archiver.setIncludeEmptyDirs( false );
346             archiver.addDirectory( m_tempDir );
347             archiver.createArchive();
348 
349             // in case we want to deploy
350             artifact.setFile( jarFile );
351         }
352         catch( ArchiverException e )
353         {
354             throw new MojoExecutionException( "Unable to archive archetype directory", e );
355         }
356         catch( IOException e )
357         {
358             throw new MojoExecutionException( "I/O error archiving archetype directory", e );
359         }
360 
361         try
362         {
363             // install fragment to local repository
364             installer.install( jarFile, artifact, repo );
365         }
366         catch( ArtifactInstallationException e )
367         {
368             throw new MojoExecutionException( "Unable to install archetype in local repository", e );
369         }
370 
371         try
372         {
373             // clean-up after ourselves...
374             FileUtils.deleteDirectory( m_tempDir );
375         }
376         catch( IOException e )
377         {
378             throw new MojoExecutionException( "Unable to remove archetype files", e );
379         }
380     }

382     /**
383      * Translate file content to work with Pax-Construct v2 archetype processing
384      * 
385      * @param text original text
386      * @return translated text
387      */
388     private String translateTextFile( String text )
389     {
390         String content = text;
391 
392         // protect special content from accidental replacement
393         content = StringUtils.replace( content, "$", "${dollar}" );
394         content = StringUtils.replace( content, "#", "${hash}" );
395 
396         content = "#set( $dollar = '$' )" + System.getProperty( "line.separator" ) + content;
397         content = "#set( $hash = '#' )" + System.getProperty( "line.separator" ) + content;
398 
399         // standard archetype translation
400         content = StringUtils.replace( content, m_namespace, "${package}" );
401 
402         // Pax-Construct v1 => v2 translation
403         content = StringUtils.replace( content, "bundle.package", "bundle.namespace" );
404         content = StringUtils.replace( content, "jar.groupId", "wrapped.groupId" );
405         content = StringUtils.replace( content, "jar.artifactId", "wrapped.artifactId" );
406         content = StringUtils.replace( content, "jar.version", "wrapped.version" );
407 
408         return content;
409     }

411     /**
412      * Translate file content to work with Pax-Construct v2 archetype processing
413      * 
414      * @param fromDir original base directory
415      * @param originalPath original path
416      * @param toDir target base directory
417      * @param mappedPath mapped path
418      * @return NO_SUCH_FILE, BINARY_FILE or TEXT_FILE
419      */
420     private int translateFile( File fromDir, String originalPath, File toDir, String mappedPath )
421     {
422         File from = new File( fromDir, originalPath );
423         FileOutputStream out = null;
424 
425         try
426         {
427             FileInputStream in = new FileInputStream( from );
428             byte[] raw = IOUtil.toByteArray( in );
429             IOUtil.close( in );
430 
431             String text;
432             try
433             {
434                 text = IOUtil.toString( raw );
435             }
436             catch( IOException e )
437             {
438                 text = "";
439             }
440 
441             if( Arrays.equals( text.getBytes(), raw ) )
442             {
443                 // text files can be mapped to new paths
444                 File file = new File( toDir, mappedPath );
445                 file.getParentFile().mkdirs();
446 
447                 out = new FileOutputStream( file );
448                 IOUtil.copy( translateTextFile( text ), out );
449 
450                 return TEXT_FILE;
451             }
452 
453             // binary files cannot be mapped to new paths
454             File file = new File( toDir, originalPath );
455             file.getParentFile().mkdirs();
456 
457             out = new FileOutputStream( file );
458             IOUtil.copy( raw, out );
459 
460             return BINARY_FILE;
461         }
462         catch( IOException e )
463         {
464             System.err.println( "WARNING: unable to clone " + from );
465             return NO_SUCH_FILE;
466         }
467         finally
468         {
469             IOUtil.close( out );
470         }
471     }
