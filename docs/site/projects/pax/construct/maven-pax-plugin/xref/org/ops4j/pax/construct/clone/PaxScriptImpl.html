93      /**
94       * Represents a simple command option, such as -w foo or -Dz=bar
95       */
96      private class Option extends Flag
97      {
98          /**
99           * Option separator
100          */
101         private char m_separator;
102 
103         /**
104          * Specified value
105          */
106         private String m_value;
107 
108         /**
109          * @param option name of the option
110          * @param separator separator between option and value
111          * @param value specified value
112          */
113         public Option( String option, char separator, String value )
114         {
115             super( option );
116             m_separator = separator;
117             m_value = value;
118         }
119 
120         /**
121          * {@inheritDoc}
122          */
123         public String toString()
124         {
125             if( ' ' == m_separator )
126             {
127                 // just protect the value (works on Windows and UNIX)
128                 return super.toString() + m_separator + '\"' + m_value + '\"';
129             }
130 
131             // protect entire option (works on Windows and UNIX)
132             return '\"' + super.toString() + m_separator + m_value + '\"';
133         }
134     }

241         /**
242          * Builder implementation for Maven specific options
243          */
244         class MavenOption
245             implements MavenOptionBuilder
246         {
247             /**
248              * {@inheritDoc}
249              */
250             public MavenOptionBuilder flag( String flag )
251             {
252                 getMvnOptions().add( new Flag( 'D' + flag ) );
253                 return this;
254             }
255 
256             /**
257              * {@inheritDoc}
258              */
259             public MavenOptionBuilder option( String option, String value )
260             {
261                 if( "targetDirectory".equals( option ) )
262                 {
263                     setTargetDir( value );
264                 }
265 
266                 getMvnOptions().add( new Option( 'D' + option, '=', value ) );
267                 return this;
268             }
269         }
270 
271         /**
272          * {@inheritDoc}
273          */
274         public String toString()
275         {
276             StringBuffer buf = new StringBuffer( "pax-" + m_name );
277 
278             for( Iterator i = m_paxOptions.iterator(); i.hasNext(); )
279             {
280                 buf.append( ' ' );
281                 buf.append( i.next() );
282             }
283 
284             // need option separator?
285             if( !m_mvnOptions.isEmpty() )
286             {
287                 buf.append( " --" );
288             }
289 
290             for( Iterator i = m_mvnOptions.iterator(); i.hasNext(); )
291             {
292                 buf.append( ' ' );
293                 buf.append( i.next() );
294             }
295 
296             return buf.toString();
297         }

300     /**
301      * Sort Pax-Construct commands by their target directory, so projects are created before they are used
302      */
303     static class ByTargetDir
304         implements Comparator
305     {
306         /**
307          * {@inheritDoc}
308          */
309         public int compare( Object lhs, Object rhs )
310         {
311             if( lhs instanceof PaxCommand )
312             {
313                 if( rhs instanceof PaxCommand )
314                 {
315                     return compare( (PaxCommand) lhs, (PaxCommand) rhs );
316                 }
317 
318                 return 1;
319             }
320             else if( rhs instanceof PaxCommand )
321             {
322                 return -1;
323             }
324 
325             return 0;
326         }
327 
328         /**
329          * Compare two Pax commands to decide which should go first
330          * 
331          * @param lhsCommand Pax command
332          * @param rhsCommand Pax command
333          * @return -1 (lhs before rhs), 0 (lhs same time as rhs), or 1 (lhs after rhs)
334          */
335         private int compare( PaxCommand lhsCommand, PaxCommand rhsCommand )
336         {
337             boolean lhsIsImport = IMPORT_BUNDLE.equals( lhsCommand.getName() );
338             boolean rhsIsImport = IMPORT_BUNDLE.equals( rhsCommand.getName() );
339 
340             if( lhsIsImport && !rhsIsImport )
341             {
342                 return 1;
343             }
344             else if( !lhsIsImport && rhsIsImport )
345             {
346                 return -1;
347             }
348             else
349             {
350                 return lhsCommand.getTargetDir().compareTo( rhsCommand.getTargetDir() );
351             }
352         }
353     }

355     /**
356      * {@inheritDoc}
357      */
358     public void write( File scriptFile )
359         throws IOException
360     {
361         scriptFile.getParentFile().mkdirs();
362 
363         BufferedWriter writer = new BufferedWriter( StreamFactory.newPlatformWriter( scriptFile ) );
364 
365         // standard UNIX shell script header
366         if( !scriptFile.getName().endsWith( ".bat" ) )
367         {
368             writer.write( "#!/bin/sh" );
369             writer.newLine();
370             writer.newLine();
371         }
372 
373         // Sort so projects are created before their bundles
374         Collections.sort( m_commands, new ByTargetDir() );
375 
376         for( Iterator i = m_commands.iterator(); i.hasNext(); )
377         {
378             if( scriptFile.getName().endsWith( ".bat" ) )
379             {
380                 // need this in batch files
381                 writer.write( "call " );
382             }
383 
384             writer.write( i.next().toString() );
385             writer.newLine();
386         }
387 
388         IOUtil.close( writer );
389     }
