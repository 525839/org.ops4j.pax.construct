359     public static boolean isBundleProject( MavenProject project, ArtifactResolver resolver, List remoteRepos,
360         ArtifactRepository localRepo, boolean testMetadata )
361     {
362         String packaging = project.getPackaging();
363         if( packaging != null && packaging.indexOf( "bundle" ) >= 0 )
364         {
365             return true;

368         return isBundleArtifact( project.getArtifact(), resolver, remoteRepos, localRepo, testMetadata );

371     /**
372      * @param artifact Maven artifact
373      * @param resolver artifact resolver
374      * @param remoteRepos sequence of remote repositories
375      * @param localRepo local Maven repository
376      * @param testMetadata check jar manifest for OSGi attributes if true
377      * @return true if this is an OSGi bundle artifact, otherwise false
378      */
379     public static boolean isBundleArtifact( Artifact artifact, ArtifactResolver resolver, List remoteRepos,
380         ArtifactRepository localRepo, boolean testMetadata )
381     {
382         String type = artifact.getType();
383         if( null != type && type.indexOf( "bundle" ) >= 0 )
384         {
385             return true;
386         }
387         else if( !testMetadata || !downloadFile( artifact, resolver, remoteRepos, localRepo ) )
388         {
389             return false;
390         }
391 
392         try
393         {
394             return isBundleArtifact( new JarFile( artifact.getFile() ).getManifest() );
395         }
396         catch( IOException e )
397         {
398             return false;
399         }
400     }

402     /**
403      * @param manifest jar manifest, possibly null
404      * @return true if this is an OSGi bundle artifact, otherwise false
405      */
406     private static boolean isBundleArtifact( Manifest manifest )
407     {
408         if( null == manifest )
409         {
410             return false;
411         }
412 
413         Attributes mainAttributes = manifest.getMainAttributes();
414 
415         return mainAttributes.getValue( "Bundle-SymbolicName" ) != null
416             || mainAttributes.getValue( "Bundle-Name" ) != null;
417     }

433      * Look for the artifact in local and remote Maven repositories
434      * 
435      * @param artifact Maven artifact
436      * @param resolver artifact resolver
437      * @param remoteRepos sequence of remote repositories
438      * @param localRepo local Maven repository

441     public static boolean downloadFile( Artifact artifact, ArtifactResolver resolver, List remoteRepos,
442         ArtifactRepository localRepo )
443     {
444         if( artifact.getFile() == null || !artifact.getFile().exists() )
445         {
446             try
447             {
448                 resolver.resolve( artifact, remoteRepos, localRepo );
449             }
450             catch( AbstractArtifactResolutionException e )
451             {
452                 return false;
453             }
454             catch( NullPointerException e )
455             {
456                 return false;
457             }
458         }
459 
460         return true;
461     }

463     /**
464      * Try to combine overlapping group and artifact identifiers to remove duplicate elements
465      * 
466      * @param groupId project group id
467      * @param artifactId project artifact id
468      * @return the combined group and artifact sequence
469      */
470     public static String getCompoundId( String groupId, String artifactId )
471     {
472         // treat any dashes like dots
473         String lhs = groupId.replace( '-', '.' );
474         String rhs = artifactId.replace( '-', '.' );
475 
476         // simple common prefix check
477         if( rhs.equals( lhs ) || rhs.startsWith( lhs + '.' ) )
478         {
479             return artifactId;
480         }
481 
482         rhs = '.' + rhs; // optimization when testing for overlap
483 
484         // check for overlapping segments by repeated chopping of artifactId
485         for( int i = rhs.length(); i > 0; i = rhs.lastIndexOf( '.', i - 1 ) )
486         {
487             if( lhs.endsWith( rhs.substring( 0, i ) ) )
488             {
489                 if( rhs.length() == i )
490                 {
491                     return groupId;
492                 }
493 
494                 return groupId + '.' + artifactId.substring( i );
495             }
496         }
497 
498         // no common segments, so append
499         return groupId + '.' + artifactId;
500     }

503      * Find the symbolic (meta) Maven version, such as 1.0-SNAPSHOT
504      * 
505      * @param artifact Maven artifact
506      * @return meta version for the artifact
507      */
508     public static String getMetaVersion( Artifact artifact )
509     {
510         if( artifact.isSnapshot() )
511         {
512             try
513             {
514                 return artifact.getSelectedVersion().toString();
515             }
516             catch( OverConstrainedVersionException e )
517             {
518                 return artifact.getVersion();
519             }
520             catch( NullPointerException e )
521             {
522                 return artifact.getVersion();
523             }
524         }
525 
526         return artifact.getVersion();
527     }

538     /**
539      * @param artifact Maven artifact
540      * @param source metadata source
541      * @param remoteRepos sequence of remote repositories
542      * @param localRepo local Maven repository
543      * @param range acceptable versions
544      * @return the release version if available, otherwise throws {@link MojoExecutionException}
545      * @throws MojoExecutionException
546      */
547     public static String getReleaseVersion( Artifact artifact, ArtifactMetadataSource source, List remoteRepos,
548         ArtifactRepository localRepo, VersionRange range )
549         throws MojoExecutionException
550     {
551         try
552         {
553             List versions = source.retrieveAvailableVersions( artifact, localRepo, remoteRepos );
554             ArtifactVersion releaseVersion = getLatestReleaseInRange( versions, range );
555             if( null == releaseVersion )
556             {
557                 throw new MojoExecutionException( "Unable to find release version for " + artifact );
558             }
559             return releaseVersion.toString();
560         }
561         catch( ArtifactMetadataRetrievalException e )
562         {
563             throw new MojoExecutionException( "Unable to find artifact " + artifact );
564         }
565     }

567     /**
568      * @param versions list of available versions
569      * @param range acceptable range of versions
570      * @return latest acceptable release, otherwise null
571      */
572     private static ArtifactVersion getLatestReleaseInRange( List versions, VersionRange range )
573     {
574         final ArtifactVersion baseline = new DefaultArtifactVersion( "0" );
575 
576         ArtifactVersion releaseVersion = baseline;
577         for( Iterator i = versions.iterator(); i.hasNext(); )
578         {
579             ArtifactVersion v = (ArtifactVersion) i.next();
580             if( range != null && !range.containsVersion( v ) )
581             {
582                 continue; // ignore this version
583             }
584             else if( releaseVersion.compareTo( v ) <= 0 && !ArtifactUtils.isSnapshot( v.toString() ) )
585             {
586                 releaseVersion = v;
587             }
588         }
589 
590         if( baseline == releaseVersion )
591         {
592             return null;
593         }
594         return releaseVersion;
595     }
