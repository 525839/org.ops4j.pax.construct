160     /**
161      * {@inheritDoc}
162      */
163     public void execute()
164         throws MojoExecutionException
165     {
166         // general purpose Pax-Construct script
167         PaxScript buildScript = new PaxScriptImpl();
168 
169         m_bundleProjectMap = new HashMap();
170         m_majorProjectMap = new HashMap();
171 
172         for( Iterator i = m_reactorProjects.iterator(); i.hasNext(); )
173         {
174             // potential project to be converted / captured
175             MavenProject project = (MavenProject) i.next();
176             String packaging = project.getPackaging();
177 
178             if( "bundle".equals( packaging ) )
179             {
180                 handleBundleProject( buildScript, project );
181             }
182             else if( "pom".equals( packaging ) )
183             {
184                 if( isMajorProject( project ) )
185                 {
186                     markMajorProject( project );
187                 }
188                 else
189                 {
190                     handleBundleImport( buildScript, project );
191                 }
192             }
193             // else handled by the major project(s)
194         }
195 
196         // collect up all non-bundle projects and settings
197         archiveMajorProjects( buildScript );
198         writePlatformScripts( buildScript );
199     }

201     /**
202      * Write out various platform-specific scripts based on the abstract build script
203      * 
204      * @param script build script
205      */
206     private void writePlatformScripts( PaxScript script )
207     {
208         String cloneId = PomUtils.getCompoundId( m_rootGroupId, m_rootArtifactId );
209         String scriptName = "pax-clone-" + cloneId;
210 
211         File winScript = new File( m_tempdir, scriptName + ".bat" );
212         File nixScript = new File( m_tempdir, scriptName );
213 
214         getLog().info( "" );
215         getLog().info( "SUCCESSFULLY CLONED " + cloneId );
216         getLog().info( "" );
217 
218         try
219         {
220             getLog().info( "Saving UNIX shell script " + nixScript );
221             script.write( nixScript );
222         }
223         catch( IOException e )
224         {
225             getLog().warn( "Unable to write " + nixScript );
226         }
227 
228         try
229         {
230             getLog().info( "Saving Windows batch file " + winScript );
231             script.write( winScript );
232         }
233         catch( IOException e )
234         {
235             getLog().warn( "Unable to write " + winScript );
236         }
237 
238         getLog().info( "" );
239         getLog().info( "FRAGMENT DIRECTORY " + getFragmentDir() );
240     }

242     /**
243      * Analyze bundle project and determine if any pax-create-bundle or pax-wrap-jar calls are needed
244      * 
245      * @param script build script
246      * @param project Maven bundle project
247      * @throws MojoExecutionException
248      */
249     private void handleBundleProject( PaxScript script, MavenProject project )
250         throws MojoExecutionException
251     {
252         PaxCommandBuilder command;
253         String bundleName;
254         String contents;
255 
256         String namespace = findBundleNamespace( project );
257         if( null != namespace )
258         {
259             bundleName = project.getArtifactId();
260 
261             command = script.call( PaxScript.CREATE_BUNDLE );
262             command.option( 'p', namespace );
263             command.option( 'n', bundleName );
264             command.option( 'v', project.getVersion() );
265         }
266         else
267         {
268             Dependency wrappee = findWrappee( project );
269             if( wrappee != null )
270             {
271                 command = script.call( PaxScript.WRAP_JAR );
272                 command.option( 'g', wrappee.getGroupId() );
273                 command.option( 'a', wrappee.getArtifactId() );
274                 command.option( 'v', wrappee.getVersion() );
275 
276                 if( repair )
277                 {
278                     // this is expected to be the generated bundle name
279                     bundleName = PomUtils.getCompoundId( wrappee.getGroupId(), wrappee.getArtifactId() );
280 
281                     // detect if we need to add the version back later on...
282                     if( project.getArtifactId().endsWith( wrappee.getVersion() ) )
283                     {
284                         command.maven().flag( "addVersion" );
285                     }
286                 }
287                 else
288                 {
289                     bundleName = project.getArtifactId();
290 
291                     // need to retain the old name and version settings
292                     command.maven().option( "bundleName", bundleName );
293                     command.maven().option( "bundleVersion", project.getVersion() );
294                 }
295             }
296             else
297             {
298                 getLog().warn( "Unable to clone bundle project " + project.getId() );
299                 return;
300             }
301         }
302 
303         Pom pom = null;
304 
305         if( repair )
306         {
307             // fix references to local bundles by re-importing
308             pom = repairBundleImports( script, project );
309         }
310         else
311         {
312             // need to keep old groupId intact (name is already retained)
313             command.maven().option( "bundleGroupId", project.getGroupId() );
314         }
315 
316         // customized sources, POMs, Bnd instructions
317         contents = createBundleArchetype( project, namespace, pom );
318         command.maven().option( "contents", contents );
319 
320         // enable overwrite
321         command.flag( 'o' );
322 
323         setTargetDirectory( command, project.getBasedir().getParentFile() );
324         addHandledProject( project, bundleName );
325     }

327     /**
328      * Analyze POM project and determine if any pax-import-bundle calls are needed
329      * 
330      * @param script build script
331      * @param project Maven POM project
332      */
333     private void handleBundleImport( PaxScript script, MavenProject project )
334     {
335         Dependency importee = findImportee( project );
336         if( importee != null )
337         {
338             PaxCommandBuilder command;
339 
340             command = script.call( PaxScript.IMPORT_BUNDLE );
341             command.option( 'g', importee.getGroupId() );
342             command.option( 'a', importee.getArtifactId() );
343             command.option( 'v', importee.getVersion() );
344 
345             // enable overwrite
346             command.flag( 'o' );
347 
348             // imported bundles now in provision POM
349             setTargetDirectory( command, m_basedir );
350             addHandledProject( project, null );
351         }
352         // else handled by the major project
353     }

355     /**
356      * Analyze the position of this project in the tree, as not all projects need their own distinct set of "poms"
357      * 
358      * @param project Maven POM project
359      * @return true if this project requires a pax-create-project call
360      */
361     private boolean isMajorProject( MavenProject project )
362     {
363         if( project.isExecutionRoot() )
364         {
365             // top-most project
366             return true;
367         }
368 
369         // disconnected project, or project with its own set of "poms" where settings can be customized
370         return ( null == project.getParent() || new File( project.getBasedir(), "poms" ).isDirectory() );
371     }

373     /**
374      * Analyze bundle project to see if it actually just wraps another artifact
375      * 
376      * @param project Maven bundle project
377      * @return wrapped artifact, null if it isn't a wrapper project
378      */
379     private Dependency findWrappee( MavenProject project )
380     {
381         Properties properties = project.getProperties();
382         Dependency wrappee = new Dependency();
383 
384         // Pax-Construct v2
385         wrappee.setGroupId( properties.getProperty( "wrapped.groupId" ) );
386         wrappee.setArtifactId( properties.getProperty( "wrapped.artifactId" ) );
387         wrappee.setVersion( properties.getProperty( "wrapped.version" ) );
388 
389         if( null == wrappee.getArtifactId() )
390         {
391             // original Pax-Construct
392             wrappee.setGroupId( properties.getProperty( "jar.groupId" ) );
393             wrappee.setArtifactId( properties.getProperty( "jar.artifactId" ) );
394             wrappee.setVersion( properties.getProperty( "jar.version" ) );
395 
396             if( null == wrappee.getArtifactId() )
397             {
398                 // has someone customized their wrapper?
399                 return findCustomizedWrappee( project );
400             }
401         }
402 
403         return wrappee;
404     }

406     /**
407      * Analyze project structure to try to deduce if this really is a wrapper
408      * 
409      * @param project Maven bundle project
410      * @return wrapped artifact, null if it isn't a wrapper project
411      */
412     private Dependency findCustomizedWrappee( MavenProject project )
413     {
414         List dependencies = project.getDependencies();
415         String sourcePath = project.getBuild().getSourceDirectory();
416 
417         // assume first dependency is wrapped artifact (unless has source)
418         if( dependencies.size() > 0 && !new File( sourcePath ).exists() )
419         {
420             return (Dependency) dependencies.get( 0 );
421         }
422 
423         return null;
424     }

450     /**
451      * Analyze bundle project to find the primary namespace it provides
452      * 
453      * @param project Maven project
454      * @return primary Java namespace
455      */
456     private String findBundleNamespace( MavenProject project )
457     {
458         Properties properties = project.getProperties();
459 
460         // Pax-Construct v2
461         String namespace = properties.getProperty( "bundle.namespace" );
462         if( null == namespace )
463         {
464             // original Pax-Construct
465             namespace = properties.getProperty( "bundle.package" );
466             String sourcePath = project.getBuild().getSourceDirectory();
467             if( null == namespace && new File( sourcePath ).exists() )
468             {
469                 namespace = findPrimaryPackage( sourcePath );
470             }
471         }
472 
473         return namespace;
474     }

476     /**
477      * Find the most likely candidate for the primary Java package
478      * 
479      * @param dir source directory
480      * @return primary Java package
481      */
482     private String findPrimaryPackage( String dir )
483     {
484         String[] pathInclude = new String[]
485         {
486             // look for bundle activators or any internal source code
487             "**/*Activator.java", "**/internal/*.java", "**/impl/*.java"
488         };
489 
490         DirectoryScanner scanner = new DirectoryScanner();
491 
492         scanner.setIncludes( pathInclude );
493         scanner.setFollowSymlinks( false );
494         scanner.addDefaultExcludes();
495         scanner.setBasedir( dir );
496 
497         scanner.scan();
498 
499         String javaFile = null;
500 
501         String[] candidates = scanner.getIncludedFiles();
502         for( int i = 0; i < scanner.getIncludedFiles().length; i++ )
503         {
504             // favour bundle activators over everything
505             if( null == javaFile || candidates[i].endsWith( "Activator.java" ) )
506             {
507                 javaFile = candidates[i];
508             }
509             // otherwise pick the internal package with the shortest path
510             else if( !javaFile.endsWith( "Activator.java" ) && candidates[i].length() < javaFile.length() )
511             {
512                 javaFile = candidates[i];
513             }
514         }
515 
516         return getJavaNamespace( javaFile );
517     }

519     /**
520      * Convert source code location into dotted Java namespace
521      * 
522      * @param javaFile source location
523      * @return Java namespace
524      */
525     private String getJavaNamespace( String javaFile )
526     {
527         if( null == javaFile )
528         {
529             return null;
530         }
531 
532         // strip the classname and any internal package
533         File packageDir = new File( javaFile ).getParentFile();
534         if( "internal".equals( packageDir.getName() ) || "impl".equals( packageDir.getName() ) )
535         {
536             packageDir = packageDir.getParentFile();
537         }
538 
539         // standard slashes to dots conversion
540         return packageDir.getPath().replaceAll( "[/\\\\]+", "." );
541     }

543     /**
544      * Set the directory where the Pax-Construct command should be run
545      * 
546      * @param command Pax-Construct command
547      * @param targetDir target directory
548      */
549     private void setTargetDirectory( PaxCommandBuilder command, File targetDir )
550     {
551         String[] pivot = DirUtils.calculateRelativePath( m_basedir.getParentFile(), targetDir );
552         if( pivot != null && pivot[2].length() > 0 )
553         {
554             // fix path to use the correct artifactId, in case directory tree has been renamed
555             String relativePath = StringUtils.replaceOnce( pivot[2], m_basedir.getName(), m_rootArtifactId );
556             command.maven().option( "targetDirectory", relativePath );
557         }
558     }

560     /**
561      * Create a new archetype for a bundle project, with potentially customized POM and Bnd settings
562      * 
563      * @param project Maven project
564      * @param namespace Java namespace, can be null
565      * @param pom customized Maven project model
566      * @return clause identifying the archetype fragment
567      * @throws MojoExecutionException
568      */
569     private String createBundleArchetype( MavenProject project, String namespace, Pom pom )
570         throws MojoExecutionException
571     {
572         File baseDir = project.getBasedir();
573 
574         ArchetypeFragment fragment = new ArchetypeFragment( getFragmentDir(), namespace );
575 
576         fragment.addPom( baseDir, pom );
577         fragment.addResources( baseDir, "osgi.bnd", false );
578 
579         if( null != namespace )
580         {
581             fragment.addSources( baseDir, project.getBuild().getSourceDirectory(), false );
582             fragment.addSources( baseDir, project.getBuild().getTestSourceDirectory(), true );
583         }
584 
585         for( Iterator i = project.getResources().iterator(); i.hasNext(); )
586         {
587             Resource r = (Resource) i.next();
588             fragment.addResources( baseDir, r.getDirectory(), r.getIncludes(), r.getExcludes(), false );
589         }
590         for( Iterator i = project.getTestResources().iterator(); i.hasNext(); )
591         {
592             Resource r = (Resource) i.next();
593             fragment.addResources( baseDir, r.getDirectory(), r.getIncludes(), r.getExcludes(), true );
594         }
595 
596         // should we look for additional non-Maven resources?
597         if( includeResources != null && includeResources.length() > 0 )
598         {
599             String[] includes = includeResources.split( "," );
600             fragment.addResources( baseDir, baseDir.getPath(), Arrays.asList( includes ), null, false );
601         }
602 
603         // archetype must use different id
604         String groupId = project.getGroupId();
605         String artifactId = project.getArtifactId() + "-archetype";
606         String version = project.getVersion();
607 
608         // install in local repository so it's accessible to the Pax-Construct scripts later on
609         Artifact artifact = m_factory.createBuildArtifact( groupId, artifactId, version, "jar" );
610         fragment.install( artifact, newJarArchiver(), m_installer, m_localRepo );
611 
612         // attach in case of later deployment
613         project.addAttachedArtifact( artifact );
614 
615         return groupId + ':' + artifactId + ':' + version;
616     }

618     /**
619      * Attempt to repair bundle imports by replacing them with pax-import-bundle commands
620      * 
621      * @param script build script
622      * @param project Maven project
623      * @return customized Maven project model
624      */
625     private Pom repairBundleImports( PaxScript script, MavenProject project )
626     {
627         Pom pom;
628         try
629         {
630             File tempFile = new File( m_tempdir, "pom.temp" );
631             FileUtils.copyFile( project.getFile(), tempFile );
632             pom = PomUtils.readPom( tempFile );
633             tempFile.deleteOnExit();
634         }
635         catch( IOException e )
636         {
637             pom = null;
638         }
639 
640         for( Iterator i = project.getDependencies().iterator(); i.hasNext(); )
641         {
642             Dependency dependency = (Dependency) i.next();
643 
644             String bundleId = dependency.getGroupId() + ':' + dependency.getArtifactId();
645             String bundleName = (String) m_bundleProjectMap.get( bundleId );
646 
647             // is this a local bundle project?
648             if( null != bundleName )
649             {
650                 PaxCommandBuilder command;
651 
652                 command = script.call( PaxScript.IMPORT_BUNDLE );
653                 command.option( 'a', bundleName );
654 
655                 // enable overwrite
656                 command.flag( 'o' );
657 
658                 // only apply to the importing bundle project
659                 setTargetDirectory( command, project.getBasedir() );
660 
661                 if( null != pom )
662                 {
663                     pom.removeDependency( dependency );
664                 }
665             }
666         }
667 
668         try
669         {
670             if( null != pom )
671             {
672                 pom.write();
673             }
674             return pom;
675         }
676         catch( IOException e )
677         {
678             return null;
679         }
680     }

682     /**
683      * Go through local project tree looking for non-bundle files to archive
684      * 
685      * @param script build script
686      * @throws MojoExecutionException
687      */
688     private void archiveMajorProjects( PaxScript script )
689         throws MojoExecutionException
690     {
691         PaxCommandBuilder command = null;
692         List resourcePaths = new ArrayList();
693 
694         Pom majorPom = null;
695         for( Iterator i = new PomIterator( m_basedir, true ); i.hasNext(); )
696         {
697             Pom pom = (Pom) i.next();
698 
699             if( m_majorProjectMap.containsKey( getCloneId( pom ) ) )
700             {
701                 // flush previous major archetype to local repo
702                 createMajorArchetype( majorPom, command, resourcePaths );
703 
704                 command = script.call( PaxScript.CREATE_PROJECT );
705                 command.option( 'g', pom.getGroupId() );
706                 command.option( 'a', pom.getArtifactId() );
707                 command.option( 'v', pom.getVersion() );
708 
709                 // enable overwrite
710                 command.flag( 'o' );
711 
712                 setTargetDirectory( command, pom.getBasedir().getParentFile() );
713 
714                 resourcePaths.clear();
715                 majorPom = pom;
716             }
717             else if( !m_bundleProjectMap.containsKey( getCloneId( pom ) ) )
718             {
719                 if( "pom".equals( pom.getPackaging() ) )
720                 {
721                     // only copy the POM, not subfolders
722                     resourcePaths.add( pom.getFile() );
723                 }
724                 else
725                 {
726                     // add the entire non-bundle project
727                     resourcePaths.add( pom.getBasedir() );
728                 }
729             }
730         }
731 
732         // flush previous major archetype to local repo
733         createMajorArchetype( majorPom, command, resourcePaths );
734     }

736     /**
737      * Archive all the selected resources under a single Maven archetype
738      * 
739      * @param majorPom containing Maven project
740      * @param command create-project command
741      * @param resourcePaths selected resources
742      * @throws MojoExecutionException
743      */
744     private void createMajorArchetype( Pom majorPom, PaxCommandBuilder command, List resourcePaths )
745         throws MojoExecutionException
746     {
747         // nothing to do!
748         if( null == majorPom )
749         {
750             return;
751         }
752 
753         ArchetypeFragment fragment = new ArchetypeFragment( getFragmentDir(), null );
754         fragment.addPom( majorPom.getBasedir(), null );
755 
756         for( Iterator i = resourcePaths.iterator(); i.hasNext(); )
757         {
758             fragment.addResources( majorPom.getBasedir(), i.next().toString(), false );
759         }
760 
761         // archetype must use different id
762         String groupId = majorPom.getGroupId();
763         String artifactId = majorPom.getArtifactId() + "-archetype";
764         String version = majorPom.getVersion();
765 
766         // install in local repository so it's accessible to the Pax-Construct scripts later on
767         Artifact artifact = m_factory.createBuildArtifact( groupId, artifactId, version, "jar" );
768         fragment.install( artifact, newJarArchiver(), m_installer, m_localRepo );
769         String fragmentId = groupId + ':' + artifactId + ':' + version;
770 
771         // customized POMs and non-bundle projects
772         command.maven().option( "contents", fragmentId );
773 
774         MavenProject project = (MavenProject) m_majorProjectMap.get( getCloneId( majorPom ) );
775         if( null != project )
776         {
777             // attach in case of later deployment
778             project.addAttachedArtifact( artifact );
779         }
780     }
