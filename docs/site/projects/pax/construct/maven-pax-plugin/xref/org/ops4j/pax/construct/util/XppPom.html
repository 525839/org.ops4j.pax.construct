58      /**
59       * Read Maven project details from existing file
60       * 
61       * @param pomFile XML file containing Maven project model
62       * @throws IOException
63       */
64      public XppPom( File pomFile )
65          throws IOException
66      {
67          // protect against changes in working directory
68          m_file = DirUtils.resolveFile( pomFile, true );
69  
70          try
71          {
72              XmlPullParser parser = RoundTripXml.createParser();
73              Reader reader = StreamFactory.newXmlReader( m_file );
74              parser.setInput( reader );
75  
76              m_pom = Xpp3DomBuilder.build( parser, false );
77  
78              IOUtil.close( reader );
79          }
80          catch( XmlPullParserException e )
81          {
82              throw new IOException( e.getLocalizedMessage() );
83          }
84      }

86      /**
87       * Create blank Maven project module
88       * 
89       * @param pomFile XML file, may or may not exist
90       * @param groupId project group id
91       * @param artifactId project artifact id
92       */
93      public XppPom( File pomFile, String groupId, String artifactId )
94      {
95          // protect against changes in working directory
96          m_file = DirUtils.resolveFile( pomFile, true );
97  
98          m_pom = new Xpp3Dom( "project" );
99  
100         // standard header cruft
101         m_pom.setAttribute( "xmlns", "http://maven.apache.org/POM/4.0.0" );
102         m_pom.setAttribute( "xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance" );
103         m_pom.setAttribute( "xsi:schemaLocation",
104             "http://maven.apache.org/POM/4.0.0 http://maven.apache.org/maven-v4_0_0.xsd" );
105 
106         Xpp3DomMap.putValue( m_pom, "modelVersion", "4.0.0" );
107         Xpp3DomMap.putValue( m_pom, "groupId", groupId );
108         Xpp3DomMap.putValue( m_pom, "artifactId", artifactId );
109         Xpp3DomMap.putValue( m_pom, "name", "" );
110         Xpp3DomMap.putValue( m_pom, "packaging", "pom" );
111 
112         m_file.getParentFile().mkdirs();
113     }

124     /**
125      * {@inheritDoc}
126      */
127     public String getParentId()
128     {
129         Xpp3Dom parent = m_pom.getChild( "parent" );
130         if( null == parent )
131         {
132             return null;
133         }
134 
135         Xpp3Dom groupId = parent.getChild( "groupId" );
136         Xpp3Dom artifactId = parent.getChild( "artifactId" );
137         Xpp3Dom version = parent.getChild( "version" );
138 
139         // assume that the parent has pom packaging (seems reasonable assumption)
140         return groupId.getValue() + ':' + artifactId.getValue() + ":pom:" + version.getValue();
141     }

146     public String getGroupId()
147     {
148         Xpp3Dom groupId = m_pom.getChild( "groupId" );
149         Xpp3Dom parent = m_pom.getChild( "parent" );
150         if( null == groupId && null != parent )
151         {
152             // inherit group from parent element
153             groupId = parent.getChild( "groupId" );
154         }
155         if( null == groupId )
156         {
157             return null;
158         }
159         return groupId.getValue();
160     }

359     /**
360      * {@inheritDoc}
361      */
362     public void addRepository( Repository repository, boolean snapshots, boolean releases, boolean overwrite,
363         boolean pluginRepo )
364         throws ExistingElementException
365     {
366         final String listName;
367         final String elemName;
368 
369         if( pluginRepo )
370         {
371             listName = "pluginRepositories";
372             elemName = "pluginRepository";
373         }
374         else
375         {
376             listName = "repositories";
377             elemName = "repository";
378         }
379 
380         String id = repository.getId();
381         String url = repository.getUrl();
382 
383         String xpath = listName + '/' + elemName + "[id='" + id + "' or url='" + url + "']";
384 
385         // clear old elements when overwriting
386         if( findChildren( xpath, overwrite ) && !overwrite )
387         {
388             throw new ExistingElementException( elemName );
389         }
390 
391         Xpp3DomMap repo = new Xpp3DomMap( elemName );
392         repo.putValue( "id", id );
393         repo.putValue( "url", url );
394 
395         if( !snapshots )
396         {
397             Xpp3DomMap snapshotFlag = new Xpp3DomMap( "snapshots" );
398             snapshotFlag.putValue( "enabled", "false" );
399             repo.addChild( snapshotFlag );
400         }
401 
402         if( !releases )
403         {
404             Xpp3DomMap releaseFlag = new Xpp3DomMap( "releases" );
405             releaseFlag.putValue( "enabled", "false" );
406             repo.addChild( releaseFlag );
407         }
408 
409         Xpp3Dom list = new Xpp3DomList( listName );
410         list.addChild( repo );
411 
412         Xpp3Dom newPom = new Xpp3Dom( "project" );
413         newPom.addChild( list );
414 
415         Xpp3DomHelper.mergeXpp3Dom( m_pom, newPom );
416     }

454     /**
455      * {@inheritDoc}
456      */
457     public void addDependency( Dependency dependency, boolean overwrite )
458         throws ExistingElementException
459     {
460         String groupId = dependency.getGroupId();
461         String artifactId = dependency.getArtifactId();
462 
463         String xpath = "dependencies/dependency[groupId='" + groupId + "' and artifactId='" + artifactId + "']";
464 
465         // clear old elements when overwriting
466         if( findChildren( xpath, overwrite ) && !overwrite )
467         {
468             throw new ExistingElementException( "dependency" );
469         }
470 
471         Xpp3DomMap dep = new Xpp3DomMap( "dependency" );
472         dep.putValue( "groupId", groupId );
473         dep.putValue( "artifactId", artifactId );
474         dep.putValue( "version", dependency.getVersion() );
475         dep.putValue( "scope", dependency.getScope() );
476 
477         String type = dependency.getType();
478         if( !"jar".equals( type ) )
479         {
480             // jar is the default type
481             dep.putValue( "type", type );
482         }
483 
484         if( dependency.isOptional() )
485         {
486             dep.putValue( "optional", "true" );
487         }
488 
489         Xpp3Dom list = new Xpp3DomList( "dependencies" );
490         list.addChild( dep );
491 
492         Xpp3Dom newPom = new Xpp3Dom( "project" );
493         newPom.addChild( list );
494 
495         Xpp3DomHelper.mergeXpp3Dom( m_pom, newPom );
496     }

517     /**
518      * @param xpath simple XPATH query
519      * @param newGroupId new group id
520      * @return true if any elements were updated, otherwise false
521      */
522     private boolean updateGroupId( String xpath, String newGroupId )
523     {
524         XppPathQuery pathQuery = new XppPathQuery( xpath );
525         Xpp3Dom parent = pathQuery.queryParent( m_pom );
526         if( null == parent )
527         {
528             return false;
529         }
530 
531         int[] children = pathQuery.queryChildren( parent );
532         for( int i = 0; i < children.length; i++ )
533         {
534             Xpp3Dom group = parent.getChild( children[i] ).getChild( "groupId" );
535             if( null != group )
536             {
537                 group.setValue( newGroupId );
538             }
539         }
540         return children.length > 0;
541     }

543     /**
544      * @param xpath simple XPATH query
545      * @param newVersion new version
546      * @return true if any elements were updated, otherwise false
547      */
548     private boolean updateVersion( String xpath, String newVersion )
549     {
550         XppPathQuery pathQuery = new XppPathQuery( xpath );
551         Xpp3Dom parent = pathQuery.queryParent( m_pom );
552         if( null == parent )
553         {
554             return false;
555         }
556 
557         int[] children = pathQuery.queryChildren( parent );
558         for( int i = 0; i < children.length; i++ )
559         {
560             Xpp3Dom fragment = parent.getChild( children[i] );
561             Xpp3Dom version = fragment.getChild( "version" );
562             if( null == version )
563             {
564                 version = new Xpp3Dom( "version" );
565                 Xpp3DomList.addChild( fragment, 2, version );
566             }
567             version.setValue( newVersion );
568         }
569         return children.length > 0;
570     }

592      * {@inheritDoc}
593      */
594     public void addExclusion( String groupId, String artifactId, boolean overwrite )
595         throws ExistingElementException
596     {
597         Xpp3Dom dependencies = m_pom.getChild( "dependencies" );
598         if( null == dependencies || dependencies.getChildCount() <= 0 )
599         {
600             return; // can't exclude what isn't there!
601         }
602 
603         String exclusionPath = "dependencies/dependency/exclusions/exclusion";
604         String xpath = exclusionPath + "[groupId='" + groupId + "' and artifactId='" + artifactId + "']";
605 
606         // clear old elements when overwriting
607         if( findChildren( xpath, overwrite ) && !overwrite )
608         {
609             throw new ExistingElementException( "exclusion" );
610         }
611 
612         Xpp3DomMap exclude = new Xpp3DomMap( "exclusion" );
613         exclude.putValue( "groupId", groupId );
614         exclude.putValue( "artifactId", artifactId );
615 
616         Xpp3Dom list = new Xpp3DomList( "exclusions" );
617         list.addChild( exclude );
618 
619         Xpp3Dom newDependency = new Xpp3Dom( "dependency" );
620         newDependency.addChild( list );
621 
622         // add exclusion to top-most dependency
623         Xpp3DomHelper.mergeXpp3Dom( dependencies.getChild( 0 ), newDependency );
624     }

710     /**
711      * Merge a section of XML from another XML fragment
712      * 
713      * @param from another XML fragment
714      * @param fromSection path to XML section to merge from
715      * @param toSection path to XML section to merge into
716      * @param append when true, append instead of merging
717      */
718     private void mergeSection( Xpp3Dom from, String fromSection, String toSection, boolean append )
719     {
720         String[] fromPath = fromSection.split( "/" );
721 
722         // find source section
723         Xpp3Dom source = from;
724         for( int i = 0; i < fromPath.length; i++ )
725         {
726             source = source.getChild( fromPath[i] );
727             if( null == source )
728             {
729                 return;
730             }
731         }
732 
733         if( append )
734         {
735             Xpp3DomList.makeIntoList( source );
736         }
737 
738         Xpp3Dom project = new Xpp3Dom( "project" );
739 
740         // create skeleton template
741         Xpp3Dom skeleton = project;
742         if( toSection != null )
743         {
744             String[] toPath = toSection.split( "/" );
745             for( int i = 0; i < toPath.length; i++ )
746             {
747                 Xpp3Dom temp = new Xpp3Dom( toPath[i] );
748                 skeleton.addChild( temp );
749                 skeleton = temp;
750             }
751         }
752 
753         // add source to template
754         skeleton.addChild( source );
755 
756         m_pom = Xpp3DomHelper.mergeXpp3Dom( m_pom, project );
757     }

759     /**
760      * {@inheritDoc}
761      */
762     public void overlayDetails( Pom pom )
763     {
764         if( !( pom instanceof XppPom ) )
765         {
766             throw new IllegalArgumentException( "Unable to overlay POM type " + pom.getClass() );
767         }
768 
769         Xpp3Dom overlay = ( (XppPom) pom ).m_pom;
770         Xpp3Dom project = new Xpp3Dom( "project" );
771 
772         // record before we drop any elements
773         List newModules = pom.getModuleNames();
774 
775         // avoid corruption of key elements
776         removeProtectedElements( overlay );
777 
778         Xpp3Dom[] sections = m_pom.getChildren();
779         for( int i = 0; i < sections.length; i++ )
780         {
781             // provide basic XML framework underneath the overlay
782             if( null == overlay.getChild( sections[i].getName() ) )
783             {
784                 project.addChild( sections[i] );
785             }
786         }
787 
788         Xpp3Dom originalPom = new Xpp3Dom( m_pom );
789         m_pom = Xpp3DomHelper.mergeXpp3Dom( project, overlay );
790 
791         // we want to keep these plugins exactly as they were in the original Pax-Construct v2 POMs

932         {
933             int count = parent.getChildCount();
934

948     /**
949      * Local utility method to check for child elements based on a simple XPATH query
950      * 
951      * @param xpath simple XPATH query
952      * @param clear remove matching elements
953      * @return true if any child elements matched, otherwise false
954      */
955     private boolean findChildren( String xpath, boolean clear )
956     {
957         XppPathQuery pathQuery = new XppPathQuery( xpath );
958         Xpp3Dom parent = pathQuery.queryParent( m_pom );
959 
960         if( null == parent )
961         {
962             return false;
963         }
964 
965         int[] children = pathQuery.queryChildren( parent );
966 
967         if( clear )
968         {
969             // sort ascending order
970             Arrays.sort( children );
971 
972             // now remove in reverse in case array shrinks
973             for( int i = children.length - 1; i >= 0; i-- )
974             {
975                 parent.removeChild( children[i] );
976             }
977         }
978 
979         return children.length > 0;
980     }
