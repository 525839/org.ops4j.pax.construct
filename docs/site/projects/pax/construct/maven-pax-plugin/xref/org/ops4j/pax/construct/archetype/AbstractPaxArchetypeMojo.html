290     /**
291      * {@inheritDoc}
292      */
293     public final void execute()
294         throws MojoExecutionException
295     {
296         // to support better templates
297         VelocityBridge.setMojo( this );
298 
299         updateFields();
300         createModuleTree();
301 
302         /*
303          * support repeated creation of projects
304          */
305         do
306         {
307             scheduleCustomArchetypes();
308             updateExtensionFields();
309 
310             prepareTarget();
311             super.execute();
312             cacheSettings();
313 
314             runCustomArchetypes();
315 
316             postProcess();
317             cleanUp();
318 
319         } while( createMoreArtifacts() );
320     }

418     /**
419      * @return acceptable version range of Pax-Construct archetypes
420      */
421     private VersionRange getArchetypeVersionRange()
422     {
423         ArtifactVersion version = new DefaultArtifactVersion( pluginVersion );
424 
425         int major = version.getMajorVersion();
426         int minor = version.getMinorVersion();
427 
428         try
429         {
430             // limit version range to the same major.minor release as the current plugin
431             String spec = "[" + major + '.' + minor + ',' + major + '.' + ( 1 + minor ) + ')';
432             return VersionRange.createFromVersionSpec( spec );
433         }
434         catch( InvalidVersionSpecificationException e )
435         {
436             return null;
437         }
438     }

440     /**
441      * Attempts to find the latest released (or snapshot) archetype that's compatible with this plugin
442      * 
443      * @param groupId archetype group id
444      * @param artifactId archetype artifact id
445      * @return compatible archetype version
446      */
447     private String getArchetypeVersion( String groupId, String artifactId )
448     {
449         // special case when plugin is a snapshot - use the local archetype snapshot, if available
450         Artifact artifact = m_factory.createBuildArtifact( groupId, artifactId, pluginVersion, "jar" );
451         if( artifact.isSnapshot() && PomUtils.getFile( artifact, m_resolver, m_localRepo ) )
452         {
453             return pluginVersion;
454         }
455 
456         // limit archetypes to same major and minor version
457         VersionRange range = getArchetypeVersionRange();
458 
459         try
460         {
461             getLog().info( "Selecting latest archetype release within version range " + range );
462             return PomUtils.getReleaseVersion( artifact, m_source, Collections.EMPTY_LIST, m_localRepo, range );
463         }
464         catch( MojoExecutionException e )
465         {
466             return pluginVersion;
467         }
468     }

487     /**
488      * Lay the foundations for the new project
489      * 
490      * @throws MojoExecutionException
491      */
492     private void prepareTarget()
493         throws MojoExecutionException
494     {
495         String artifactId = (String) m_archetypeMojo.getField( "artifactId" );
496         File pomDirectory = new File( targetDirectory, artifactId );
497 
498         // support overwriting of existing projects
499         m_pomFile = new File( pomDirectory, "pom.xml" );
500         if( canOverwrite() && m_pomFile.exists() )
501         {
502             m_pomFile.delete();
503         }
504 
505         // reset trashcan
506         m_tempFiles = new FileSet();
507         m_tempFiles.setDirectory( pomDirectory.getAbsolutePath() );
508 
509         if( pomDirectory.exists() )
510         {
511             preserveExistingFiles( pomDirectory );
512         }
513         else
514         {
515             pomDirectory.mkdirs();
516         }
517 
518         if( null != m_modulesPom )
519         {
520             try
521             {
522                 // attach new project to its physical parent
523                 m_modulesPom.addModule( pomDirectory.getName(), canOverwrite() );
524                 m_modulesPom.write();
525             }
526             catch( IOException e )
527             {
528                 getLog().warn( "Unable to attach POM to existing project" );
529             }
530         }
531     }

534      * @param baseDir project base directory
535      * @throws MojoExecutionException
536      */
537     private void preserveExistingFiles( File baseDir )
538         throws MojoExecutionException
539     {
540         try
541         {
542             // exclude existing files, so we don't accidentally trash them later
543             List excludes = FileUtils.getFileNames( baseDir, null, null, false );
544             for( Iterator i = excludes.iterator(); i.hasNext(); )
545             {
546                 getLog().debug( "Preserving " + i.next() );
547             }
548             m_tempFiles.setExcludes( excludes );
549         }
550         catch( IOException e )
551         {
552             throw new MojoExecutionException( "I/O error while protecting existing files from deletion", e );
553         }
554     }

556     /**
557      * Cache the original generated files before any custom archetypes run
558      * 
559      * @throws MojoExecutionException
560      */
561     private void cacheSettings()
562         throws MojoExecutionException
563     {
564         try
565         {
566             if( null != m_modulesPom )
567             {
568                 // before caching, search for logical parent in project tree
569                 DirUtils.updateLogicalParent( m_pomFile, getParentId() );
570             }
571 
572             m_pom = PomUtils.readPom( m_pomFile );
573             if( hasCustomContent() )
574             {
575                 // allow further customization
576                 m_pom.getFile().delete();
577             }
578         }
579         catch( IOException e )
580         {
581             throw new MojoExecutionException( "I/O error reading generated Maven POM " + m_pomFile, e );
582         }
583 
584         try
585         {
586             m_bnd = BndUtils.readBnd( m_pom.getBasedir() );
587             if( hasCustomContent() )
588             {
589                 // allow further customization
590                 m_bnd.getFile().delete();
591             }
592         }
593         catch( IOException e )
594         {
595             throw new MojoExecutionException( "I/O error reading generated Bnd instructions", e );
596         }
597 
598         if( hasCustomContent() )
599         {
600             // no need to cache if not customizing
601             cacheOriginalFiles( m_pom.getBasedir() );
602         }
603     }

697     /**
698      * Add custom Maven archetypes, to be used after the main archetype has finished
699      */
700     private void scheduleCustomArchetypes()
701     {
702         m_customArchetypeIds = new ArrayList();
703 
704         // use default content
705         if( !hasCustomContent() || contents.trim().length() == 0 )
706         {
707             return;
708         }
709 
710         String[] ids = contents.split( "," );
711         for( int i = 0; i < ids.length; i++ )
712         {
713             String id = ids[i].trim();
714 
715             // handle groupId:artifactId:other:stuff
716             String[] fields = id.split( ":" );
717             if( fields.length > 2 )
718             {
719                 // fully-qualified external archetype
720                 scheduleArchetype( fields[0], fields[1], fields[2] );
721             }
722             else if( fields.length > 1 )
723             {
724                 // semi-qualified external archetype (assume groupId same as artifactId)
725                 scheduleArchetype( fields[0], fields[0], fields[1] );
726             }
727             else
728             {
729                 // internal Pax-Construct archetype (assume same version as archetype template)
730                 scheduleArchetype( PAX_CONSTRUCT_GROUP_ID, fields[0], null );
731             }
732         }
733     }

754     /**
755      * @return set of filenames that will be left at the end of this archetype cycle
756      */
757     protected final Set getFinalFilenames()
758     {
759         Set finalFiles = new HashSet();
760 
761         DirectoryScanner scanner = new DirectoryScanner();
762         scanner.setBasedir( m_tempFiles.getDirectory() );
763         scanner.setFollowSymlinks( false );
764 
765         scanner.addDefaultExcludes();
766         scanner.setExcludes( m_tempFiles.getExcludesArray() );
767         scanner.setIncludes( m_tempFiles.getIncludesArray() );
768 
769         scanner.scan();
770 
771         finalFiles.addAll( Arrays.asList( scanner.getNotIncludedFiles() ) );
772         finalFiles.addAll( Arrays.asList( scanner.getExcludedFiles() ) );
773 
774         return finalFiles;
775     }

777     /**
778      * Clean up any temporary or unnecessary files, including empty directories
779      */
780     private void cleanUp()
781     {
782         DirectoryScanner scanner = new DirectoryScanner();
783         scanner.setBasedir( m_tempFiles.getDirectory() );
784         scanner.setFollowSymlinks( false );
785 
786         scanner.addDefaultExcludes();
787         scanner.setExcludes( m_tempFiles.getExcludesArray() );
788         scanner.setIncludes( m_tempFiles.getIncludesArray() );
789 
790         scanner.scan();
791 
792         String[] discardedFiles = scanner.getIncludedFiles();
793         for( int i = 0; i < discardedFiles.length; i++ )
794         {
795             String filename = discardedFiles[i];
796             getLog().debug( "Discarding " + filename );
797             new File( scanner.getBasedir(), filename ).delete();
798         }
799 
800         // remove any empty directories after the cleanup
801         DirUtils.pruneEmptyFolders( scanner.getBasedir() );
802     }
